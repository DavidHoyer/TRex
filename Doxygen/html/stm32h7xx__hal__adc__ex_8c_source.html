459HAL_StatusTypeDef HAL_ADCEx_InjectedStart(ADC_HandleTypeDef *hadc)
460{
461 HAL_StatusTypeDef tmp_hal_status;
462 uint32_t tmp_config_injected_queue;
463 uint32_t tmp_multimode_config = LL_ADC_GetMultimode(__LL_ADC_COMMON_INSTANCE(hadc->Instance));
464
465 /* Check the parameters */
466 assert_param(IS_ADC_ALL_INSTANCE(hadc->Instance));
467
468 if (LL_ADC_INJ_IsConversionOngoing(hadc->Instance) != 0UL)
469 {
470 return HAL_BUSY;
471 }
472 else
473 {
474 /* In case of software trigger detection enabled, JQDIS must be set
475 (which can be done only if ADSTART and JADSTART are both cleared).
476 If JQDIS is not set at that point, returns an error
477 - since software trigger detection is disabled. User needs to
478 resort to HAL_ADCEx_DisableInjectedQueue() API to set JQDIS.
479 - or (if JQDIS is intentionally reset) since JEXTEN = 0 which means
480 the queue is empty */
481 tmp_config_injected_queue = READ_BIT(hadc->Instance->CFGR, ADC_CFGR_JQDIS);
482
483 if ((READ_BIT(hadc->Instance->JSQR, ADC_JSQR_JEXTEN) == 0UL)
484 && (tmp_config_injected_queue == 0UL)
485 )
486 {
487 SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_CONFIG);
488 return HAL_ERROR;
489 }
490
491 /* Process locked */
492 __HAL_LOCK(hadc);
493
494 /* Enable the ADC peripheral */
495 tmp_hal_status = ADC_Enable(hadc);
496
497 /* Start conversion if ADC is effectively enabled */
498 if (tmp_hal_status == HAL_OK)
499 {
500 /* Check if a regular conversion is ongoing */
501 if ((hadc->State & HAL_ADC_STATE_REG_BUSY) != 0UL)
502 {
503 /* Reset ADC error code field related to injected conversions only */
504 CLEAR_BIT(hadc->ErrorCode, HAL_ADC_ERROR_JQOVF);
505 }
506 else
507 {
508 /* Set ADC error code to none */
509 ADC_CLEAR_ERRORCODE(hadc);
510 }
511
512 /* Set ADC state */
513 /* - Clear state bitfield related to injected group conversion results */
514 /* - Set state bitfield related to injected operation */
515 ADC_STATE_CLR_SET(hadc->State,
516 HAL_ADC_STATE_READY | HAL_ADC_STATE_INJ_EOC,
517 HAL_ADC_STATE_INJ_BUSY);
518
519 /* Reset HAL_ADC_STATE_MULTIMODE_SLAVE bit
520 - if ADC instance is master or if multimode feature is not available
521 - if multimode setting is disabled (ADC instance slave in independent mode) */
522 if ((__LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance) == hadc->Instance)
523 || (tmp_multimode_config == LL_ADC_MULTI_INDEPENDENT)
524 )
525 {
526 CLEAR_BIT(hadc->State, HAL_ADC_STATE_MULTIMODE_SLAVE);
527 }
528
529 /* Clear ADC group injected group conversion flag */
530 /* (To ensure of no unknown state from potential previous ADC operations) */
531 __HAL_ADC_CLEAR_FLAG(hadc, (ADC_FLAG_JEOC | ADC_FLAG_JEOS));
532
533 /* Process unlocked */
534 /* Unlock before starting ADC conversions: in case of potential */
535 /* interruption, to let the process to ADC IRQ Handler. */
536 __HAL_UNLOCK(hadc);
537
538 /* Enable conversion of injected group, if automatic injected conversion */
539 /* is disabled. */
540 /* If software start has been selected, conversion starts immediately. */
541 /* If external trigger has been selected, conversion will start at next */
542 /* trigger event. */
543 /* Case of multimode enabled (when multimode feature is available): */
544 /* if ADC is slave, */
545 /* - ADC is enabled only (conversion is not started), */
546 /* - if multimode only concerns regular conversion, ADC is enabled */
547 /* and conversion is started. */
548 /* If ADC is master or independent, */
549 /* - ADC is enabled and conversion is started. */
550 if ((__LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance) == hadc->Instance)
551 || (tmp_multimode_config == LL_ADC_MULTI_INDEPENDENT)
552 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_REG_SIMULT)
553 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_REG_INTERL)
554 )
555 {
556 /* ADC instance is not a multimode slave instance with multimode injected conversions enabled */
557 if (LL_ADC_INJ_GetTrigAuto(hadc->Instance) == LL_ADC_INJ_TRIG_INDEPENDENT)
558 {
559 LL_ADC_INJ_StartConversion(hadc->Instance);
560 }
561 }
562 else
563 {
564 /* ADC instance is not a multimode slave instance with multimode injected conversions enabled */
565 SET_BIT(hadc->State, HAL_ADC_STATE_MULTIMODE_SLAVE);
566 }
567
568 }
569 else
570 {
571 /* Process unlocked */
572 __HAL_UNLOCK(hadc);
573 }
574
575 /* Return function status */
576 return tmp_hal_status;
577 }

651HAL_StatusTypeDef HAL_ADCEx_InjectedPollForConversion(ADC_HandleTypeDef *hadc, uint32_t Timeout)
652{
653 uint32_t tickstart;
654 uint32_t tmp_Flag_End;
655 uint32_t tmp_adc_inj_is_trigger_source_sw_start;
656 uint32_t tmp_adc_reg_is_trigger_source_sw_start;
657 uint32_t tmp_cfgr;
658 const ADC_TypeDef *tmpADC_Master;
659 uint32_t tmp_multimode_config = LL_ADC_GetMultimode(__LL_ADC_COMMON_INSTANCE(hadc->Instance));
660
661 /* Check the parameters */
662 assert_param(IS_ADC_ALL_INSTANCE(hadc->Instance));
663
664 /* If end of sequence selected */
665 if (hadc->Init.EOCSelection == ADC_EOC_SEQ_CONV)
666 {
667 tmp_Flag_End = ADC_FLAG_JEOS;
668 }
669 else /* end of conversion selected */
670 {
671 tmp_Flag_End = ADC_FLAG_JEOC;
672 }
673
674 /* Get timeout */
675 tickstart = HAL_GetTick();
676
677 /* Wait until End of Conversion or Sequence flag is raised */
678 while ((hadc->Instance->ISR & tmp_Flag_End) == 0UL)
679 {
680 /* Check if timeout is disabled (set to infinite wait) */
681 if (Timeout != HAL_MAX_DELAY)
682 {
683 if (((HAL_GetTick() - tickstart) > Timeout) || (Timeout == 0UL))
684 {
685 if((hadc->Instance->ISR & tmp_Flag_End) == 0UL)
686 {
687 /* Update ADC state machine to timeout */
688 SET_BIT(hadc->State, HAL_ADC_STATE_TIMEOUT);
689
690 /* Process unlocked */
691 __HAL_UNLOCK(hadc);
692
693 return HAL_TIMEOUT;
694 }
695 }
696 }
697 }
698
699 /* Retrieve ADC configuration */
700 tmp_adc_inj_is_trigger_source_sw_start = LL_ADC_INJ_IsTriggerSourceSWStart(hadc->Instance);
701 tmp_adc_reg_is_trigger_source_sw_start = LL_ADC_REG_IsTriggerSourceSWStart(hadc->Instance);
702 /* Get relevant register CFGR in ADC instance of ADC master or slave */
703 /* in function of multimode state (for devices with multimode */
704 /* available). */
705 if ((__LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance) == hadc->Instance)
706 || (tmp_multimode_config == LL_ADC_MULTI_INDEPENDENT)
707 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_REG_SIMULT)
708 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_REG_INTERL)
709 )
710 {
711 tmp_cfgr = READ_REG(hadc->Instance->CFGR);
712 }
713 else
714 {
715 tmpADC_Master = __LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance);
716 tmp_cfgr = READ_REG(tmpADC_Master->CFGR);
717 }
718
719 /* Update ADC state machine */
720 SET_BIT(hadc->State, HAL_ADC_STATE_INJ_EOC);
721
722 /* Determine whether any further conversion upcoming on group injected */
723 /* by external trigger or by automatic injected conversion */
724 /* from group regular. */
725 if ((tmp_adc_inj_is_trigger_source_sw_start != 0UL) ||
726 ((READ_BIT(tmp_cfgr, ADC_CFGR_JAUTO) == 0UL) &&
727 ((tmp_adc_reg_is_trigger_source_sw_start != 0UL) &&
728 (READ_BIT(tmp_cfgr, ADC_CFGR_CONT) == 0UL))))
729 {
730 /* Check whether end of sequence is reached */
731 if (__HAL_ADC_GET_FLAG(hadc, ADC_FLAG_JEOS))
732 {
733 /* Particular case if injected contexts queue is enabled: */
734 /* when the last context has been fully processed, JSQR is reset */
735 /* by the hardware. Even if no injected conversion is planned to come */
736 /* (queue empty, triggers are ignored), it can start again */
737 /* immediately after setting a new context (JADSTART is still set). */
738 /* Therefore, state of HAL ADC injected group is kept to busy. */
739 if (READ_BIT(tmp_cfgr, ADC_CFGR_JQM) == 0UL)
740 {
741 /* Set ADC state */
742 CLEAR_BIT(hadc->State, HAL_ADC_STATE_INJ_BUSY);
743
744 if ((hadc->State & HAL_ADC_STATE_REG_BUSY) == 0UL)
745 {
746 SET_BIT(hadc->State, HAL_ADC_STATE_READY);
747 }
748 }
749 }
750 }
751
752 /* Clear polled flag */
753 if (tmp_Flag_End == ADC_FLAG_JEOS)
754 {
755 /* Clear end of sequence JEOS flag of injected group if low power feature */
756 /* "LowPowerAutoWait " is disabled, to not interfere with this feature. */
757 /* For injected groups, no new conversion will start before JEOS is */
758 /* cleared. */
759 if (READ_BIT(tmp_cfgr, ADC_CFGR_AUTDLY) == 0UL)
760 {
761 __HAL_ADC_CLEAR_FLAG(hadc, (ADC_FLAG_JEOC | ADC_FLAG_JEOS));
762 }
763 }
764 else
765 {
766 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_JEOC);
767 }
768
769 /* Return API HAL status */
770 return HAL_OK;
771}

1138

1141
1142

1144 tmp_hal_status = ADC_ConversionStop(hadc, ADC_REGULAR_INJECTED_GROUP);
1145

1164 }

1190 __HAL_UNLOCK(hadc);
1191
1192 return HAL_ERROR;
1193 }
1194 }

1196 tmphadcSlave_conversion_on_going = LL_ADC_REG_IsConversionOngoing((&tmphadcSlave)->Instance);

1200 /* while DMA transfer is on going) */

1209 SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_DMA);
1210 }
1211
1212 /* Disable ADC overrun interrupt */
1213 __HAL_ADC_DISABLE_IT(hadc, ADC_IT_OVR);
1214

1216 /* Update "tmp_hal_status" only if DMA channel disabling passed, to keep in */
1217 /* memory a potential failing status. */
1218 if (tmp_hal_status == HAL_OK)
1219 {
1220 tmphadcSlave_disable_status = ADC_Disable(&tmphadcSlave);
1221 if ((ADC_Disable(hadc) == HAL_OK) &&
1222 (tmphadcSlave_disable_status == HAL_OK))
1223 {
1224 tmp_hal_status = HAL_OK;
1225 }
1226 }
1227 else
1228 {
1229 /* In case of error, attempt to disable ADC master and slave without status assert */
1230 (void) ADC_Disable(hadc);
1231 (void) ADC_Disable(&tmphadcSlave);
1232 }
1233
1234 /* Set ADC state (ADC master) */
1235 ADC_STATE_CLR_SET(hadc->State,
1236 HAL_ADC_STATE_REG_BUSY | HAL_ADC_STATE_INJ_BUSY,
1237 HAL_ADC_STATE_READY);
1238 }
1239
1240 /* Process unlocked */
1241 __HAL_UNLOCK(hadc);
1242
1243 /* Return function status */
1244 return tmp_hal_status;

1666 while ((LL_ADC_REG_IsConversionOngoing(hadc->Instance) == 1UL)
1667 || (tmphadcSlave_conversion_on_going == 1UL)

1669 {

1686 }

1693 /* while DMA transfer is on going) */

1695 /* with HAL_ADCEx_RegularStop_DMA() API. */
1696 tmp_hal_status = HAL_DMA_Abort(hadc->DMA_Handle);
1697
1698 /* Check if DMA channel effectively disabled */
1699 if (tmp_hal_status != HAL_OK)
1700 {
1701 /* Update ADC state machine to error */

1703 }
1704
1705 /* Disable ADC overrun interrupt */
1706 __HAL_ADC_DISABLE_IT(hadc, ADC_IT_OVR);
1707
1708 /* 2. Disable the ADC peripherals: master and slave if no injected */
1709 /* conversion is on-going. */
1710 /* Update "tmp_hal_status" only if DMA channel disabling passed, to keep in */
1711 /* memory a potential failing status. */
1712 if (tmp_hal_status == HAL_OK)
1713 {
1714 if (LL_ADC_INJ_IsConversionOngoing(hadc->Instance) == 0UL)
1715 {
1716 tmp_hal_status = ADC_Disable(hadc);
1717 if (tmp_hal_status == HAL_OK)
1718 {
1719 if (LL_ADC_INJ_IsConversionOngoing((&tmphadcSlave)->Instance) == 0UL)
1720 {
1721 tmp_hal_status = ADC_Disable(&tmphadcSlave);
1722 }
1723 }
1724 }
1725
1726 if (tmp_hal_status == HAL_OK)
1727 {
1728 /* Both Master and Slave ADC's could be disabled. Update Master State */
1729 /* Clear HAL_ADC_STATE_INJ_BUSY bit, set HAL_ADC_STATE_READY bit */

1731 }
1732 else
1733 {
1734 /* injected (Master or Slave) conversions are still on-going,
1735 no Master State change */
1736 }
1737 }
1738 }
1739
1740 /* Process unlocked */
1741 __HAL_UNLOCK(hadc);
1742
1743 /* Return function status */
1744 return tmp_hal_status;
1745}
