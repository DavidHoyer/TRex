177static result_t write(const uint8_t *buffer, size_t size) {
178 GPIO_SetValue(SensorChipSelect, true);
179 CORE_Delay(10);
180 result_t result = SPI_Write(SPI_Sensor, buffer, size);
181 CORE_Delay(10);
182 GPIO_SetValue(SensorChipSelect, false);
183
184 return result;
185}

197result_t ICM42605_Init(ICM42605_AccelFullScaleRange accelScaleRange, ICM42605_GyroFullScaleRange gyroScaleRange, ICM42605_AccelOutputDataRange accelOutputDataRange, ICM42605_GyroOutputDataRange gyroOutputDataRange) {
198
199 s_accelScaleRange = accelScaleRange;
200 s_gyroScaleRange = gyroScaleRange;
201
202 {
203 uint8_t writeBuffer[2] = { WHO_AM_I | REQUEST_RESPONSE, 0x00 };
204 uint8_t readBuffer[2] = { 0x00 };
205 R_TRY(writeRead(writeBuffer, readBuffer, 2));
206
207 if (readBuffer[1] != DEVICE_ID)
208 return RESULT_INITIALIZATION_ERROR;
209 }
210
211 // Enable gyroscope and accelerometer in low noise mode
212 {
213 uint8_t pwrMgmt0 = 0;
214 {
215 uint8_t writeBuffer[2] = { PWR_MGMT0 | REQUEST_RESPONSE, 0x00 };
216 uint8_t readBuffer[2] = { 0x00 };
217 R_TRY(writeRead(writeBuffer, readBuffer, 2));
218 pwrMgmt0 = readBuffer[1];
219 }
220
221 {
222 uint8_t writeBuffer[2] = { PWR_MGMT0, pwrMgmt0 | 0x0F };
223 R_TRY(write(writeBuffer, 2));
224 }
225 }
226
227 CORE_Delay(100);
228
229 // Set gyro scale and data rate
230 {
231 uint8_t gyroConfig0 = 0;
232 {
233 uint8_t writeBuffer[2] = { GYRO_CONFIG0 | REQUEST_RESPONSE, 0x00 };
234 uint8_t readBuffer[2] = { 0x00 };
235 R_TRY(writeRead(writeBuffer, readBuffer, 2));
236 gyroConfig0 = readBuffer[1];
237 }
238
239 {
240 uint8_t writeBuffer[2] = { GYRO_CONFIG0, gyroConfig0 | gyroOutputDataRange | (gyroScaleRange << 5) };
241 R_TRY(write(writeBuffer, 2));
242 }
243 }
244
245 // Set accel scale and data rate
246 {
247 uint8_t accelConfig0 = 0;
248 {
249 uint8_t writeBuffer[2] = { ACCEL_CONFIG0 | REQUEST_RESPONSE, 0x00 };
250 uint8_t readBuffer[2] = { 0x00 };
251 R_TRY(writeRead(writeBuffer, readBuffer, 2));
252 accelConfig0 = readBuffer[1];
253 }
254
255 {
256 uint8_t writeBuffer[2] = { ACCEL_CONFIG0, accelConfig0 | accelOutputDataRange | (accelScaleRange << 5) };
257 R_TRY(write(writeBuffer, 2));
258 }
259 }
260
261 // Set temperature low pass filter to a minimum and use first order filter for gyro
262 {
263 uint8_t gyroConfig1 = 0;
264 {
265 uint8_t writeBuffer[2] = { GYRO_CONFIG1 | REQUEST_RESPONSE, 0x00 };
266 uint8_t readBuffer[2] = { 0x00 };
267 R_TRY(writeRead(writeBuffer, readBuffer, 2));
268 gyroConfig1 = readBuffer[1];
269 }
270
271 {
272 uint8_t writeBuffer[2] = { GYRO_CONFIG1, gyroConfig1 | 0x0D };
273 R_TRY(write(writeBuffer, 2));
274 }
275 }
276
277 // Setup interrupt pins
278 {
279 uint8_t intConfig0 = 0;
280 {
281 uint8_t writeBuffer[2] = { INT_CONFIG | REQUEST_RESPONSE, 0x00 };
282 uint8_t readBuffer[2] = { 0x00 };
283 R_TRY(writeRead(writeBuffer, readBuffer, 2));
284 intConfig0 = readBuffer[1];
285 }
286
287 {
288 uint8_t writeBuffer[2] = { INT_CONFIG, intConfig0 | 0x18 | 0x03 };
289 R_TRY(write(writeBuffer, 2));
290 }
291 }
292
293 // Enable async interrupt reset
294 {
295 uint8_t intConfig1 = 0;
296 {
297 uint8_t writeBuffer[2] = { INT_CONFIG1 | REQUEST_RESPONSE, 0x00 };
298 uint8_t readBuffer[2] = { 0x00 };
299 R_TRY(writeRead(writeBuffer, readBuffer, 2));
300 intConfig1 = readBuffer[1];
301 }
302
303 {
304 uint8_t writeBuffer[2] = { INT_CONFIG1, intConfig1 & ~0x10 };
305 R_TRY(write(writeBuffer, 2));
306 }
307 }
308
309 // Route data ready interrupt to INT1
310 {
311 uint8_t intSource0 = 0;
312 {
313 uint8_t writeBuffer[2] = { INT_SOURCE0 | REQUEST_RESPONSE, 0x00 };
314 uint8_t readBuffer[2] = { 0x00 };
315 R_TRY(writeRead(writeBuffer, readBuffer, 2));
316 intSource0 = readBuffer[1];
317 }
318
319 {
320 uint8_t writeBuffer[2] = { INT_SOURCE0, intSource0 | 0x08 };
321 R_TRY(write(writeBuffer, 2));
322 }
323 }
324
325 // Route AGC interrupt to INT2
326 {
327 uint8_t intSource3 = 0;
328 {
329 uint8_t writeBuffer[2] = { INT_SOURCE3 | REQUEST_RESPONSE, 0x00 };
330 uint8_t readBuffer[2] = { 0x00 };
331 R_TRY(writeRead(writeBuffer, readBuffer, 2));
332 intSource3 = readBuffer[1];
333 }
334
335 {
336 uint8_t writeBuffer[2] = { INT_SOURCE3, intSource3 | 0x01 };
337 R_TRY(write(writeBuffer, 2));
338 }
339 }
340
341 return RESULT_SUCCESS;
342}

344result_t ICM42605_GetRotation(vector3f_t *value) {
345 if (value == NULL)
346 return RESULT_INVALID_ARGUMENTS;
347
348 uint8_t writeBuffer[7] = { GYRO_DATA_X1 | REQUEST_RESPONSE, 0x00 };
349 uint8_t readBuffer[7] = { 0x00 };
350
351 R_TRY(writeRead(writeBuffer, readBuffer, 7));
352
353 value->x = ((float32_t)((int16_t)(readBuffer[2] | readBuffer[1] << 8)) / 0xFFFF) * (2000 / (s_gyroScaleRange + 1));
354 value->y = ((float32_t)((int16_t)(readBuffer[4] | readBuffer[3] << 8)) / 0xFFFF) * (2000 / (s_gyroScaleRange + 1));
355 value->z = ((float32_t)((int16_t)(readBuffer[6] | readBuffer[5] << 8)) / 0xFFFF) * (2000 / (s_gyroScaleRange + 1));
356
357 return RESULT_SUCCESS;
358}
