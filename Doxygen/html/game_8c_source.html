186event_T CheckEvent(void)
187{
188 event_T event;
189 LCD_TouchPosition_t touchPosition = {0, 0};
190 static char flag = FALSE;
191
192 if (R_SUCCESS(LCD_TouchGetPosition(&touchPosition)))
193 {
194 if(touchPosition.x == 0 && touchPosition.y == 480){
195 event.eventFlag = FALSE;
196 flag = FALSE;
197 return event;
198 }
199
200 if(flag == TRUE){
201 event.eventFlag = FALSE;
202 return event;
203 }
204
205 event.eventFlag = TRUE;
206 event.x = touchPosition.x;
207 event.y = touchPosition.y;
208 flag = TRUE;
209 }
210
211 return event;

278
279 if(tRexDirection == MOVE_NONE)
280 return;
281

371// Given three collinear points p, q, r, the function checks if
372// point q lies on line segment 'pr'
373static bool onSegment(Point p, Point q, Point r)
374{
375 if (q.x <= fmax(p.x, r.x) && q.x >= fmin(p.x, r.x) &&
376 q.y <= fmax(p.y, r.y) && q.y >= fmin(p.y, r.y))
377 return true;
378
379 return false;
380}
381
382// To find orientation of ordered triplet (p, q, r).
383// The function returns following values
384// 0 --> p, q and r are collinear
385// 1 --> Clockwise
386// 2 --> Counterclockwise
387static int orientation(Point p, Point q, Point r)
388{
389 // See https://www.geeksforgeeks.org/orientation-3-ordered-points/
390 // for details of below formula.
391 int val = (q.y - p.y) * (r.x - q.x) -
392 (q.x - p.x) * (r.y - q.y);
393
394 if (val == 0) return 0; // collinear
395
396 return (val > 0)? 1: 2; // clock or counterclock wise
397}

399// The main function that returns true if line segment 'p1q1' and 'p2q2' intersect.
400//bool doIntersect(Point p1, Point q1, Point p2, Point q2)
401static int doIntersect(uint16_t A1_x, uint16_t A1_y, //Line 1 Pos A
402 uint16_t A2_x, uint16_t A2_y, //Line 1 Pos B
403 uint16_t B1_x, uint16_t B1_y, //Line 2 Pos A
404 uint16_t B2_x, uint16_t B2_y) //Line 2 Pos B
405{
406 Point p1 = {A1_x, A1_y};
407 Point q1 = {A2_x, A2_y};
408 Point p2 = {B1_x, B1_y};
409 Point q2 = {B2_x, B2_y};
410
411 // Find the four orientations needed for general and special cases
412 int o1 = orientation(p1, q1, p2);
413 int o2 = orientation(p1, q1, q2);
414 int o3 = orientation(p2, q2, p1);
415 int o4 = orientation(p2, q2, q1);
416
417 // General case
418 if (o1 != o2 && o3 != o4)
419 return true;
420
421 // Special Cases
422 // p1, q1 and p2 are collinear and p2 lies on segment p1q1
423 if (o1 == 0 && onSegment(p1, p2, q1))
424 return true;
425
426 // p1, q1 and q2 are collinear and q2 lies on segment p1q1
427 if (o2 == 0 && onSegment(p1, q2, q1))
428 return true;
429
430 // p2, q2 and p1 are collinear and p1 lies on segment p2q2
431 if (o3 == 0 && onSegment(p2, p1, q2))
432 return true;
433
434 // p2, q2 and q1 are collinear and q1 lies on segment p2q2
435 if (o4 == 0 && onSegment(p2, q1, q2))
436 return true;
437
438 return false; // Doesn't fall in any of the above cases
439}

442//*** Checks if the T-Rex is colliding with one of the obstacles ***
443//*********************************************************************
444char CheckCollision(void){
445 if(tRexBmp.visible == FALSE)
446 return FALSE;
447
448 for(uint16_t i = 0; i < OBSTACLES_NUMBER; i++){
449 if(obstacleBmp[i].visible == FALSE)
450 continue;
451
452 //Check if the image borders of the T-Rex and the obstacles are colliding
453 if (tRexBmp.x + tRexBmp.w < obstacleBmp[i].x ||
454 tRexBmp.x > obstacleBmp[i].x + obstacleBmp[i].w ||
455 tRexBmp.y + tRexBmp.h < obstacleBmp[i].y ||
456 tRexBmp.y > obstacleBmp[i].y + obstacleBmp[i].h)
457 {
458 continue;
459 }
460
461 node_t *TRexPixelA = tRexBmp.head; //Point 1 of T-Rex Border
462 node_t *TRexPixelB = TRexPixelA->next; //Point 2 of T-Rex Border
463 node_t *ObstPixelA = obstacleBmp[0].head; //Point 1 of Obstacle Border
464 node_t *ObstPixelB = ObstPixelA->next; //Point 2 of Obstacle Border
465
466 while(ObstPixelA != NULL && ObstPixelB != NULL){
467 while(TRexPixelA != NULL && TRexPixelB != NULL){
468 //check for a collision between Lines
469 if(doIntersect( obstacleBmp[i].x + ObstPixelA->x, obstacleBmp[i].y + ObstPixelA->y, //Line 1 Pos A
470 obstacleBmp[i].x + ObstPixelB->x, obstacleBmp[i].y + ObstPixelB->y, //Line 1 Pos B
471 tRexBmp.x + TRexPixelA->x, tRexBmp.y + TRexPixelA->y, //Line 2 Pos A
472 tRexBmp.x + TRexPixelB->x, tRexBmp.y + TRexPixelB->y //Line 2 Pos B
473 ) == true)
474 {
475 return true;
476 }
477
478 //--- Next line of the T-Rex border
479 TRexPixelA = TRexPixelB;
480 TRexPixelB = TRexPixelA->next;
481 }
482
483 //--- Next line of the Obstacle border
484 ObstPixelA = ObstPixelB;
485 ObstPixelB = ObstPixelA->next;
486
487 //Start again on T-Rex border
488 TRexPixelA = tRexBmp.head;
489 TRexPixelB = TRexPixelA->next;
490 }
491 }
492 return false;
493}
