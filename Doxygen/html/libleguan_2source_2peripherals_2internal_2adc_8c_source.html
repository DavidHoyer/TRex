72result_t ADC_StartMeasurement(ADC_Input_t input) {
73 /* Switch channel */
74 ADC_ChannelConfTypeDef channelConfig = { 0 };
75
76 switch (input) {
77 case ADC1_14: channelConfig.Channel = ADC_CHANNEL_14; break;
78 case ADC1_15: channelConfig.Channel = ADC_CHANNEL_15; break;
79 case ADC1_16: channelConfig.Channel = ADC_CHANNEL_16; break;
80 case ADC2_3: channelConfig.Channel = ADC_CHANNEL_3; break;
81
82 case Potentiometer:
83 case ADC3_8:
84 channelConfig.Channel = ADC_CHANNEL_8;
85 break;
86 default: return RESULT_INVALID_ARGUMENTS;
87 }
88
89 channelConfig.Rank = ADC_REGULAR_RANK_1;
90 channelConfig.SamplingTime = ADC_SAMPLETIME_8CYCLES_5;
91
92 ADC_HandleTypeDef *handle = GetHandle(input);
93 if (HAL_ADC_ConfigChannel(handle, &channelConfig) != HAL_OK)
94 return RESULT_ADC_MEASUREMENT_ERROR;
95
96 /* Start conversion */
97 if (HAL_ADC_Start(handle) != HAL_OK)
98 return RESULT_ADC_MEASUREMENT_ERROR;
99
100 return RESULT_SUCCESS;
101}

107result_t ADC_GetValue(ADC_Input_t input, float32_t *result) {
108 ADC_HandleTypeDef *handle = GetHandle(input);
109
110 if (HAL_ADC_PollForConversion(handle, INT32_MAX) != HAL_OK)
111 return RESULT_ADC_INVALID_VALUE;
112
113 if (result == NULL)
114 return RESULT_INVALID_ARGUMENTS;
115
116 *result = (float32_t)(HAL_ADC_GetValue(handle)) / (1U << GetResolution(input)) * ADC_REFERENCE_VOLTAGE;
117
118 return RESULT_SUCCESS;
119}
