46static result_t Set7SegDigit(uint8_t digit, char value) {
47 /* There's only four 7 segment displays, bail if out of range */
48 if (digit > 3)
49 return RESULT_INVALID_ARGUMENTS;
50
51 switch(value){
52 case 0: FPGA->seven_segment.digits[digit] = 0x3F; break;
53 case 1: FPGA->seven_segment.digits[digit] = 0x06; break;
54 case 2: FPGA->seven_segment.digits[digit] = 0x5B; break;
55 case 3: FPGA->seven_segment.digits[digit] = 0x4F; break;
56 case 4: FPGA->seven_segment.digits[digit] = 0x66; break;
57 case 5: FPGA->seven_segment.digits[digit] = 0x6D; break;
58 case 6: FPGA->seven_segment.digits[digit] = 0x7D; break;
59 case 7: FPGA->seven_segment.digits[digit] = 0x07; break;
60 case 8: FPGA->seven_segment.digits[digit] = 0x7F; break;
61 case 9: FPGA->seven_segment.digits[digit] = 0x6F; break;
62 case 0x0A: FPGA->seven_segment.digits[digit] = 0x77; break;
63 case 0x0B: FPGA->seven_segment.digits[digit] = 0x7C; break;
64 case 0x0C: FPGA->seven_segment.digits[digit] = 0x39; break;
65 case 0x0D: FPGA->seven_segment.digits[digit] = 0x5E; break;
66 case 0x0E: FPGA->seven_segment.digits[digit] = 0x79; break;
67 case 0x0F: FPGA->seven_segment.digits[digit] = 0x71; break;
68 case '-': FPGA->seven_segment.digits[digit] = 0x40; break;
69 case ' ': FPGA->seven_segment.digits[digit] = 0x00; break;
70 default: FPGA->seven_segment.digits[digit] = 0x00; break;
71 }
72
73 return RESULT_SUCCESS;
74}

76result_t FPGA_7SegDisplayDecimal(uint16_t value) {
77 /* Maximum 4 digit decimal value is 9999 */
78 if (value > 9999) value = 9999;
79
80 /* Turn every digit into bcd */
81 uint16_t bcd = 0;
82 bcd |= MATH_BinaryToBCD((value / 1000) % 10) << 12;
83 bcd |= MATH_BinaryToBCD((value / 100) % 10) << 8;
84 bcd |= MATH_BinaryToBCD((value / 10) % 10) << 4;
85 bcd |= MATH_BinaryToBCD((value / 1) % 10) << 0;
86
87 FPGA->seven_segment.dp.all = 0b0000;
88 FPGA->seven_segment.bcd.all = bcd;
89
90 return RESULT_SUCCESS;
91}

93result_t FPGA_7SegDisplayFloat(float32_t value, uint8_t precision) {
94 /* There's only 4 displays. Decimal precision higher than 3 can't be displayed */
95 if (precision > 3)
96 return RESULT_INVALID_ARGUMENTS;
97
98 bool negative = value < 0;
99
100 if (negative) {
101 if (precision == 0)
102 return RESULT_INVALID_ARGUMENTS;
103
104 precision--;
105 value *= -1;
106 }
107
108 FPGA->seven_segment.dp.all = 0b0001 << precision;
109
110 /* Move all relevant digits above decimal point */
111 for (uint8_t i = 0; i < precision; i++)
112 value *= 10;
113
114 /* Turn every digit into bcd */
115 uint16_t bcd = 0;
116 bcd |= MATH_BinaryToBCD((int32_t)(value / 1000) % 10) << 12;
117 bcd |= MATH_BinaryToBCD((int32_t)(value / 100) % 10) << 8;
118 bcd |= MATH_BinaryToBCD((int32_t)(value / 10) % 10) << 4;
119 bcd |= MATH_BinaryToBCD((int32_t)(value / 1) % 10) << 0;
120
121 FPGA->seven_segment.bcd.all = bcd;
122
123 /* Display negative sign if number was negative */
124 if (negative)
125 Set7SegDigit(3, '-');
126
127 return RESULT_SUCCESS;
128}
