871 /* Flush register JSQR: reset the queue sequencer when injected */
872 /* queue sequencer is enabled and ADC disabled. */

874 /* internally disabled just after the completion of the last valid */
875 /* injected sequence. */

893 /* and disable return status. In case of status fail, attempt to */

896 /* system RCC hard reset. */

1406HAL_StatusTypeDef HAL_ADC_Start(ADC_HandleTypeDef *hadc)
1407{
1408 HAL_StatusTypeDef tmp_hal_status;
1409 const ADC_TypeDef *tmpADC_Master;
1410 uint32_t tmp_multimode_config = LL_ADC_GetMultimode(__LL_ADC_COMMON_INSTANCE(hadc->Instance));
1411
1412 /* Check the parameters */
1413 assert_param(IS_ADC_ALL_INSTANCE(hadc->Instance));
1414
1415 /* Perform ADC enable and conversion start if no conversion is on going */
1416 if (LL_ADC_REG_IsConversionOngoing(hadc->Instance) == 0UL)
1417 {
1418 /* Process locked */
1419 __HAL_LOCK(hadc);
1420
1421 /* Enable the ADC peripheral */
1422 tmp_hal_status = ADC_Enable(hadc);
1423
1424 /* Start conversion if ADC is effectively enabled */
1425 if (tmp_hal_status == HAL_OK)
1426 {
1427 /* Set ADC state */
1428 /* - Clear state bitfield related to regular group conversion results */
1429 /* - Set state bitfield related to regular operation */
1430 ADC_STATE_CLR_SET(hadc->State,
1431 HAL_ADC_STATE_READY | HAL_ADC_STATE_REG_EOC | HAL_ADC_STATE_REG_OVR | HAL_ADC_STATE_REG_EOSMP,
1432 HAL_ADC_STATE_REG_BUSY);
1433
1434 /* Reset HAL_ADC_STATE_MULTIMODE_SLAVE bit
1435 - if ADC instance is master or if multimode feature is not available
1436 - if multimode setting is disabled (ADC instance slave in independent mode) */
1437 if ((__LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance) == hadc->Instance)
1438 || (tmp_multimode_config == LL_ADC_MULTI_INDEPENDENT)
1439 )
1440 {
1441 CLEAR_BIT(hadc->State, HAL_ADC_STATE_MULTIMODE_SLAVE);
1442 }
1443
1444 /* Set ADC error code */
1445 /* Check if a conversion is on going on ADC group injected */
1446 if (HAL_IS_BIT_SET(hadc->State, HAL_ADC_STATE_INJ_BUSY))
1447 {
1448 /* Reset ADC error code fields related to regular conversions only */
1449 CLEAR_BIT(hadc->ErrorCode, (HAL_ADC_ERROR_OVR | HAL_ADC_ERROR_DMA));
1450 }
1451 else
1452 {
1453 /* Reset all ADC error code fields */
1454 ADC_CLEAR_ERRORCODE(hadc);
1455 }
1456
1457 /* Clear ADC group regular conversion flag and overrun flag */
1458 /* (To ensure of no unknown state from potential previous ADC operations) */
1459 __HAL_ADC_CLEAR_FLAG(hadc, (ADC_FLAG_EOC | ADC_FLAG_EOS | ADC_FLAG_OVR));
1460
1461 /* Process unlocked */
1462 /* Unlock before starting ADC conversions: in case of potential */
1463 /* interruption, to let the process to ADC IRQ Handler. */
1464 __HAL_UNLOCK(hadc);
1465
1466 /* Enable conversion of regular group. */
1467 /* If software start has been selected, conversion starts immediately. */
1468 /* If external trigger has been selected, conversion will start at next */
1469 /* trigger event. */
1470 /* Case of multimode enabled (when multimode feature is available): */
1471 /* - if ADC is slave and dual regular conversions are enabled, ADC is */
1472 /* enabled only (conversion is not started), */
1473 /* - if ADC is master, ADC is enabled and conversion is started. */
1474 if ((__LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance) == hadc->Instance)
1475 || (tmp_multimode_config == LL_ADC_MULTI_INDEPENDENT)
1476 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_INJ_SIMULT)
1477 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_INJ_ALTERN)
1478 )
1479 {
1480 /* ADC instance is not a multimode slave instance with multimode regular conversions enabled */
1481 if (READ_BIT(hadc->Instance->CFGR, ADC_CFGR_JAUTO) != 0UL)
1482 {
1483 ADC_STATE_CLR_SET(hadc->State, HAL_ADC_STATE_INJ_EOC, HAL_ADC_STATE_INJ_BUSY);
1484 }
1485
1486 /* Start ADC group regular conversion */
1487 LL_ADC_REG_StartConversion(hadc->Instance);
1488 }
1489 else
1490 {
1491 /* ADC instance is a multimode slave instance with multimode regular conversions enabled */
1492 SET_BIT(hadc->State, HAL_ADC_STATE_MULTIMODE_SLAVE);
1493 /* if Master ADC JAUTO bit is set, update Slave State in setting
1494 HAL_ADC_STATE_INJ_BUSY bit and in resetting HAL_ADC_STATE_INJ_EOC bit */
1495 tmpADC_Master = __LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance);
1496 if (READ_BIT(tmpADC_Master->CFGR, ADC_CFGR_JAUTO) != 0UL)
1497 {
1498 ADC_STATE_CLR_SET(hadc->State, HAL_ADC_STATE_INJ_EOC, HAL_ADC_STATE_INJ_BUSY);
1499 }
1500
1501 }
1502 }
1503 else
1504 {
1505 /* Process unlocked */
1506 __HAL_UNLOCK(hadc);
1507 }
1508 }
1509 else
1510 {
1511 tmp_hal_status = HAL_BUSY;
1512 }
1513
1514 /* Return function status */
1515 return tmp_hal_status;
1516}

1581HAL_StatusTypeDef HAL_ADC_PollForConversion(ADC_HandleTypeDef *hadc, uint32_t Timeout)
1582{
1583 uint32_t tickstart;
1584 uint32_t tmp_Flag_End;
1585 uint32_t tmp_cfgr;
1586 const ADC_TypeDef *tmpADC_Master;
1587 uint32_t tmp_multimode_config = LL_ADC_GetMultimode(__LL_ADC_COMMON_INSTANCE(hadc->Instance));
1588
1589 /* Check the parameters */
1590 assert_param(IS_ADC_ALL_INSTANCE(hadc->Instance));
1591
1592 /* If end of conversion selected to end of sequence conversions */
1593 if (hadc->Init.EOCSelection == ADC_EOC_SEQ_CONV)
1594 {
1595 tmp_Flag_End = ADC_FLAG_EOS;
1596 }
1597 /* If end of conversion selected to end of unitary conversion */
1598 else /* ADC_EOC_SINGLE_CONV */
1599 {
1600 /* Verification that ADC configuration is compliant with polling for */
1601 /* each conversion: */
1602 /* Particular case is ADC configured in DMA mode and ADC sequencer with */
1603 /* several ranks and polling for end of each conversion. */
1604 /* For code simplicity sake, this particular case is generalized to */
1605 /* ADC configured in DMA mode and and polling for end of each conversion. */
1606 if ((tmp_multimode_config == LL_ADC_MULTI_INDEPENDENT)
1607 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_INJ_SIMULT)
1608 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_INJ_ALTERN)
1609 )
1610 {
1611 /* Check DMNGT bit in handle ADC CFGR register */
1612 if (READ_BIT(hadc->Instance->CFGR, ADC_CFGR_DMNGT_0) != 0UL)
1613 {
1614 SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_CONFIG);
1615 return HAL_ERROR;
1616 }
1617 else
1618 {
1619 tmp_Flag_End = (ADC_FLAG_EOC);
1620 }
1621 }
1622 else
1623 {
1624 /* Check ADC DMA mode in multimode on ADC group regular */
1625 if (LL_ADC_GetMultiDMATransfer(__LL_ADC_COMMON_INSTANCE(hadc->Instance)) != LL_ADC_MULTI_REG_DMA_EACH_ADC)
1626 {
1627 SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_CONFIG);
1628 return HAL_ERROR;
1629 }
1630 else
1631 {
1632 tmp_Flag_End = (ADC_FLAG_EOC);
1633 }
1634 }
1635 }
1636
1637 /* Get tick count */
1638 tickstart = HAL_GetTick();
1639
1640 /* Wait until End of unitary conversion or sequence conversions flag is raised */
1641 while ((hadc->Instance->ISR & tmp_Flag_End) == 0UL)
1642 {
1643 /* Check if timeout is disabled (set to infinite wait) */
1644 if (Timeout != HAL_MAX_DELAY)
1645 {
1646 if (((HAL_GetTick() - tickstart) > Timeout) || (Timeout == 0UL))
1647 {
1648 /* New check to avoid false timeout detection in case of preemption */
1649 if((hadc->Instance->ISR & tmp_Flag_End) == 0UL)
1650 {
1651 /* Update ADC state machine to timeout */
1652 SET_BIT(hadc->State, HAL_ADC_STATE_TIMEOUT);
1653
1654 /* Process unlocked */
1655 __HAL_UNLOCK(hadc);
1656
1657 return HAL_TIMEOUT;
1658 }
1659 }
1660 }
1661 }
1662
1663 /* Update ADC state machine */
1664 SET_BIT(hadc->State, HAL_ADC_STATE_REG_EOC);
1665
1666 /* Determine whether any further conversion upcoming on group regular */
1667 /* by external trigger, continuous mode or scan sequence on going. */
1668 if ((LL_ADC_REG_IsTriggerSourceSWStart(hadc->Instance) != 0UL)
1669 && (hadc->Init.ContinuousConvMode == DISABLE)
1670 )
1671 {
1672 /* Check whether end of sequence is reached */
1673 if (__HAL_ADC_GET_FLAG(hadc, ADC_FLAG_EOS))
1674 {
1675 /* Set ADC state */
1676 CLEAR_BIT(hadc->State, HAL_ADC_STATE_REG_BUSY);
1677
1678 if ((hadc->State & HAL_ADC_STATE_INJ_BUSY) == 0UL)
1679 {
1680 SET_BIT(hadc->State, HAL_ADC_STATE_READY);
1681 }
1682 }
1683 }
1684

1750 assert_param(IS_ADC_EVENT_TYPE(EventType));

1752 /* Get tick count */
1753 tickstart = HAL_GetTick();

1760 {

1771
1772 return HAL_TIMEOUT;
1773 }
1774 }
1775 }
1776 }

1797 /* " if ((HAL_ADC_GetState(hadc1) & HAL_ADC_STATE_AWD3) != 0UL) " */
1798
1799 /* Check analog watchdog 1 flag */
1800 case ADC_AWD_EVENT:
1801 /* Set ADC state */
1802 SET_BIT(hadc->State, HAL_ADC_STATE_AWD1);
1803
1804 /* Clear ADC analog watchdog flag */
1805 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_AWD1);
1806
1807 break;
1808
1809 /* Check analog watchdog 2 flag */
1810 case ADC_AWD2_EVENT:

1812 SET_BIT(hadc->State, HAL_ADC_STATE_AWD2);
1813
1814 /* Clear ADC analog watchdog flag */
1815 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_AWD2);
1816
1817 break;
1818
1819 /* Check analog watchdog 3 flag */
1820 case ADC_AWD3_EVENT:

1823
1824 /* Clear ADC analog watchdog flag */
1825 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_AWD3);
1826
1827 break;
1828
1829 /* Injected context queue overflow event */
1830 case ADC_JQOVF_EVENT:

1832 SET_BIT(hadc->State, HAL_ADC_STATE_INJ_JQOVF);
1833
1834 /* Set ADC error code to Injected context queue overflow */

1836

1840 break;
1841
1842 /* Overrun event */
1843 default: /* Case ADC_OVR_EVENT */

1845 /* considered as an error. */
1846 /* (cf ref manual "Managing conversions without using the DMA and without */
1847 /* overrun ") */
1848 if (hadc->Init.Overrun == ADC_OVR_DATA_PRESERVED)
1849 {
1850 /* Set ADC state */
1851 SET_BIT(hadc->State, HAL_ADC_STATE_REG_OVR);
1852
1853 /* Set ADC error code to overrun */
1854 SET_BIT(hadc->ErrorCode, HAL_ADC_ERROR_OVR);
1855 }
1856 else
1857 {

1859 otherwise, data register is potentially overwritten by new converted data as soon
1860 as OVR is cleared. */
1861 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_OVR);
1862 }
1863 break;
1864 }
1865
1866 /* Return function status */
1867 return HAL_OK;
1868}

1893HAL_StatusTypeDef HAL_ADC_Start_IT(ADC_HandleTypeDef *hadc)

1918 HAL_ADC_STATE_READY | HAL_ADC_STATE_REG_EOC | HAL_ADC_STATE_REG_OVR | HAL_ADC_STATE_REG_EOSMP,

1929 }

1933 if ((hadc->State & HAL_ADC_STATE_INJ_BUSY) != 0UL)

1945 /* (To ensure of no unknown state from potential previous ADC operations) */

1950 /* interruption, to let the process to ADC IRQ Handler. */

1975 }

1979 /* If external trigger has been selected, conversion will start at next */
1980 /* trigger event. */

1983 /* enabled only (conversion is not started), */

1986 || (tmp_multimode_config == LL_ADC_MULTI_INDEPENDENT)

2004 __HAL_ADC_ENABLE_IT(hadc, ADC_IT_JEOS);
2005 break;
2006 /* case ADC_EOC_SINGLE_CONV */
2007 default:
2008 __HAL_ADC_DISABLE_IT(hadc, ADC_IT_JEOS);
2009 __HAL_ADC_ENABLE_IT(hadc, ADC_IT_JEOC);
2010 break;
2011 }
2012 }
2013

2016 }
2017 else
2018 {

2022 are enabled nevertheless (for same reason as above) */
2023 tmpADC_Master = __LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance);
2024 if (READ_BIT(tmpADC_Master->CFGR, ADC_CFGR_JAUTO) != 0UL)
2025 {
2026 /* First, update Slave State in setting HAL_ADC_STATE_INJ_BUSY bit
2027 and in resetting HAL_ADC_STATE_INJ_EOC bit */
2028 ADC_STATE_CLR_SET(hadc->State, HAL_ADC_STATE_INJ_EOC, HAL_ADC_STATE_INJ_BUSY);
2029 /* Next, set Slave injected interruptions */
2030 switch (hadc->Init.EOCSelection)
2031 {
2032 case ADC_EOC_SEQ_CONV:
2033 __HAL_ADC_DISABLE_IT(hadc, ADC_IT_JEOC);
2034 __HAL_ADC_ENABLE_IT(hadc, ADC_IT_JEOS);
2035 break;
2036 /* case ADC_EOC_SINGLE_CONV */
2037 default:
2038 __HAL_ADC_DISABLE_IT(hadc, ADC_IT_JEOS);
2039 __HAL_ADC_ENABLE_IT(hadc, ADC_IT_JEOC);
2040 break;
2041 }
2042 }
2043 }
2044 }
2045 else
2046 {
2047 /* Process unlocked */
2048 __HAL_UNLOCK(hadc);
2049 }
2050
2051 }
2052 else
2053 {
2054 tmp_hal_status = HAL_BUSY;
2055 }
2056
2057 /* Return function status */
2058 return tmp_hal_status;

2407 if ((((tmp_isr & ADC_FLAG_EOC) == ADC_FLAG_EOC) && ((tmp_ier & ADC_IT_EOC) == ADC_IT_EOC)) ||
2408 (((tmp_isr & ADC_FLAG_EOS) == ADC_FLAG_EOS) && ((tmp_ier & ADC_IT_EOS) == ADC_IT_EOS)))
2409 {
2410 /* Update state machine on conversion status if not in error state */
2411 if ((hadc->State & HAL_ADC_STATE_ERROR_INTERNAL) == 0UL)
2412 {
2413 /* Set ADC state */
2414 SET_BIT(hadc->State, HAL_ADC_STATE_REG_EOC);
2415 }
2416
2417 /* Determine whether any further conversion upcoming on group regular */
2418 /* by external trigger, continuous mode or scan sequence on going */
2419 /* to disable interruption. */
2420 if (LL_ADC_REG_IsTriggerSourceSWStart(hadc->Instance) != 0UL)
2421 {
2422 /* Get relevant register CFGR in ADC instance of ADC master or slave */
2423 /* in function of multimode state (for devices with multimode */
2424 /* available). */
2425 if ((__LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance) == hadc->Instance)
2426 || (tmp_multimode_config == LL_ADC_MULTI_INDEPENDENT)
2427 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_INJ_SIMULT)
2428 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_INJ_ALTERN)
2429 )
2430 {
2431 /* check CONT bit directly in handle ADC CFGR register */
2432 tmp_cfgr = READ_REG(hadc->Instance->CFGR);
2433 }
2434 else
2435 {
2436 /* else need to check Master ADC CONT bit */
2437 tmpADC_Master = __LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance);
2438 tmp_cfgr = READ_REG(tmpADC_Master->CFGR);
2439 }
2440
2441 /* Carry on if continuous mode is disabled */
2442 if (READ_BIT(tmp_cfgr, ADC_CFGR_CONT) != ADC_CFGR_CONT)
2443 {
2444 /* If End of Sequence is reached, disable interrupts */
2445 if (__HAL_ADC_GET_FLAG(hadc, ADC_FLAG_EOS))
2446 {
2447 /* Allowed to modify bits ADC_IT_EOC/ADC_IT_EOS only if bit */
2448 /* ADSTART==0 (no conversion on going) */
2449 if (LL_ADC_REG_IsConversionOngoing(hadc->Instance) == 0UL)
2450 {
2451 /* Disable ADC end of sequence conversion interrupt */
2452 /* Note: Overrun interrupt was enabled with EOC interrupt in */
2453 /* HAL_Start_IT(), but is not disabled here because can be used */
2454 /* by overrun IRQ process below. */
2455 __HAL_ADC_DISABLE_IT(hadc, ADC_IT_EOC | ADC_IT_EOS);
2456
2457 /* Set ADC state */
2458 CLEAR_BIT(hadc->State, HAL_ADC_STATE_REG_BUSY);
2459
2460 if ((hadc->State & HAL_ADC_STATE_INJ_BUSY) == 0UL)
2461 {
2462 SET_BIT(hadc->State, HAL_ADC_STATE_READY);
2463 }
2464 }
2465 else
2466 {
2467 /* Change ADC state to error state */
2468 SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_INTERNAL);
2469
2470 /* Set ADC error code to ADC peripheral internal error */
2471 SET_BIT(hadc->ErrorCode, HAL_ADC_ERROR_INTERNAL);
2472 }
2473 }
2474 }
2475 }
2476
2477 /* Conversion complete callback */
2478 /* Note: Into callback function "HAL_ADC_ConvCpltCallback()", */
2479 /* to determine if conversion has been triggered from EOC or EOS, */
2480 /* possibility to use: */
2481 /* " if( __HAL_ADC_GET_FLAG(&hadc, ADC_FLAG_EOS)) " */
2482#if (USE_HAL_ADC_REGISTER_CALLBACKS == 1)
2483 hadc->ConvCpltCallback(hadc);
2484#else
2485 HAL_ADC_ConvCpltCallback(hadc);
2486#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */
2487
2488 /* Clear regular group conversion flag */
2489 /* Note: in case of overrun set to ADC_OVR_DATA_PRESERVED, end of */
2490 /* conversion flags clear induces the release of the preserved data.*/
2491 /* Therefore, if the preserved data value is needed, it must be */
2492 /* read preliminarily into HAL_ADC_ConvCpltCallback(). */
2493 __HAL_ADC_CLEAR_FLAG(hadc, (ADC_FLAG_EOC | ADC_FLAG_EOS));
2494 }
2495
2496 /* ====== Check ADC group injected end of unitary conversion sequence conversions ===== */
2497 if ((((tmp_isr & ADC_FLAG_JEOC) == ADC_FLAG_JEOC) && ((tmp_ier & ADC_IT_JEOC) == ADC_IT_JEOC)) ||
2498 (((tmp_isr & ADC_FLAG_JEOS) == ADC_FLAG_JEOS) && ((tmp_ier & ADC_IT_JEOS) == ADC_IT_JEOS)))
2499 {
2500 /* Update state machine on conversion status if not in error state */
2501 if ((hadc->State & HAL_ADC_STATE_ERROR_INTERNAL) == 0UL)
2502 {
2503 /* Set ADC state */
2504 SET_BIT(hadc->State, HAL_ADC_STATE_INJ_EOC);
2505 }
2506
2507 /* Retrieve ADC configuration */
2508 tmp_adc_inj_is_trigger_source_sw_start = LL_ADC_INJ_IsTriggerSourceSWStart(hadc->Instance);
2509 tmp_adc_reg_is_trigger_source_sw_start = LL_ADC_REG_IsTriggerSourceSWStart(hadc->Instance);
2510 /* Get relevant register CFGR in ADC instance of ADC master or slave */
2511 /* in function of multimode state (for devices with multimode */
2512 /* available). */
2513 if ((__LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance) == hadc->Instance)
2514 || (tmp_multimode_config == LL_ADC_MULTI_INDEPENDENT)
2515 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_REG_SIMULT)
2516 || (tmp_multimode_config == LL_ADC_MULTI_DUAL_REG_INTERL)
2517 )
2518 {
2519 tmp_cfgr = READ_REG(hadc->Instance->CFGR);
2520 }
2521 else
2522 {
2523 tmpADC_Master = __LL_ADC_MULTI_INSTANCE_MASTER(hadc->Instance);
2524 tmp_cfgr = READ_REG(tmpADC_Master->CFGR);
2525 }
2526
2527 /* Disable interruption if no further conversion upcoming by injected */
2528 /* external trigger or by automatic injected conversion with regular */
2529 /* group having no further conversion upcoming (same conditions as */
2530 /* regular group interruption disabling above), */
2531 /* and if injected scan sequence is completed. */
2532 if ((tmp_adc_inj_is_trigger_source_sw_start != 0UL) ||
2533 ((READ_BIT(tmp_cfgr, ADC_CFGR_JAUTO) == 0UL) &&
2534 ((tmp_adc_reg_is_trigger_source_sw_start != 0UL) &&
2535 (READ_BIT(tmp_cfgr, ADC_CFGR_CONT) == 0UL))))
2536 {
2537 /* If End of Sequence is reached, disable interrupts */
2538 if (__HAL_ADC_GET_FLAG(hadc, ADC_FLAG_JEOS))
2539 {
2540 /* Particular case if injected contexts queue is enabled: */
2541 /* when the last context has been fully processed, JSQR is reset */
2542 /* by the hardware. Even if no injected conversion is planned to come */
2543 /* (queue empty, triggers are ignored), it can start again */
2544 /* immediately after setting a new context (JADSTART is still set). */
2545 /* Therefore, state of HAL ADC injected group is kept to busy. */
2546 if (READ_BIT(tmp_cfgr, ADC_CFGR_JQM) == 0UL)
2547 {
2548 /* Allowed to modify bits ADC_IT_JEOC/ADC_IT_JEOS only if bit */
2549 /* JADSTART==0 (no conversion on going) */
2550 if (LL_ADC_INJ_IsConversionOngoing(hadc->Instance) == 0UL)
2551 {
2552 /* Disable ADC end of sequence conversion interrupt */
2553 __HAL_ADC_DISABLE_IT(hadc, ADC_IT_JEOC | ADC_IT_JEOS);
2554
2555 /* Set ADC state */
2556 CLEAR_BIT(hadc->State, HAL_ADC_STATE_INJ_BUSY);
2557
2558 if ((hadc->State & HAL_ADC_STATE_REG_BUSY) == 0UL)
2559 {
2560 SET_BIT(hadc->State, HAL_ADC_STATE_READY);
2561 }
2562 }
2563 else
2564 {
2565 /* Update ADC state machine to error */
2566 SET_BIT(hadc->State, HAL_ADC_STATE_ERROR_INTERNAL);
2567
2568 /* Set ADC error code to ADC peripheral internal error */
2569 SET_BIT(hadc->ErrorCode, HAL_ADC_ERROR_INTERNAL);
2570 }
2571 }
2572 }
2573 }
2574
2575 /* Injected Conversion complete callback */
2576 /* Note: HAL_ADCEx_InjectedConvCpltCallback can resort to
2577 if( __HAL_ADC_GET_FLAG(&hadc, ADC_FLAG_JEOS)) or
2578 if( __HAL_ADC_GET_FLAG(&hadc, ADC_FLAG_JEOC)) to determine whether
2579 interruption has been triggered by end of conversion or end of
2580 sequence. */
2581#if (USE_HAL_ADC_REGISTER_CALLBACKS == 1)
2582 hadc->InjectedConvCpltCallback(hadc);
2583#else
2584 HAL_ADCEx_InjectedConvCpltCallback(hadc);
2585#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */
2586
2587 /* Clear injected group conversion flag */
2588 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_JEOC | ADC_FLAG_JEOS);
2589 }
2590
2591 /* ========== Check Analog watchdog 1 flag ========== */
2592 if (((tmp_isr & ADC_FLAG_AWD1) == ADC_FLAG_AWD1) && ((tmp_ier & ADC_IT_AWD1) == ADC_IT_AWD1))
2593 {
2594 /* Set ADC state */
2595 SET_BIT(hadc->State, HAL_ADC_STATE_AWD1);
2596
2597 /* Level out of window 1 callback */
2598#if (USE_HAL_ADC_REGISTER_CALLBACKS == 1)
2599 hadc->LevelOutOfWindowCallback(hadc);
2600#else
2601 HAL_ADC_LevelOutOfWindowCallback(hadc);
2602#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */
2603
2604 /* Clear ADC analog watchdog flag */
2605 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_AWD1);
2606 }
2607
2608 /* ========== Check analog watchdog 2 flag ========== */
2609 if (((tmp_isr & ADC_FLAG_AWD2) == ADC_FLAG_AWD2) && ((tmp_ier & ADC_IT_AWD2) == ADC_IT_AWD2))
2610 {
2611 /* Set ADC state */
2612 SET_BIT(hadc->State, HAL_ADC_STATE_AWD2);
2613
2614 /* Level out of window 2 callback */
2615#if (USE_HAL_ADC_REGISTER_CALLBACKS == 1)
2616 hadc->LevelOutOfWindow2Callback(hadc);
2617#else
2618 HAL_ADCEx_LevelOutOfWindow2Callback(hadc);
2619#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */
2620
2621 /* Clear ADC analog watchdog flag */
2622 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_AWD2);
2623 }
2624
2625 /* ========== Check analog watchdog 3 flag ========== */
2626 if (((tmp_isr & ADC_FLAG_AWD3) == ADC_FLAG_AWD3) && ((tmp_ier & ADC_IT_AWD3) == ADC_IT_AWD3))
2627 {
2628 /* Set ADC state */
2629 SET_BIT(hadc->State, HAL_ADC_STATE_AWD3);
2630
2631 /* Level out of window 3 callback */
2632#if (USE_HAL_ADC_REGISTER_CALLBACKS == 1)
2633 hadc->LevelOutOfWindow3Callback(hadc);
2634#else
2635 HAL_ADCEx_LevelOutOfWindow3Callback(hadc);
2636#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */
2637
2638 /* Clear ADC analog watchdog flag */
2639 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_AWD3);
2640 }
2641
2642 /* ========== Check Overrun flag ========== */
2643 if (((tmp_isr & ADC_FLAG_OVR) == ADC_FLAG_OVR) && ((tmp_ier & ADC_IT_OVR) == ADC_IT_OVR))
2644 {
2645 /* If overrun is set to overwrite previous data (default setting), */
2646 /* overrun event is not considered as an error. */
2647 /* (cf ref manual "Managing conversions without using the DMA and without */
2648 /* overrun ") */
2649 /* Exception for usage with DMA overrun event always considered as an */
2650 /* error. */
2651 if (hadc->Init.Overrun == ADC_OVR_DATA_PRESERVED)
2652 {
2653 overrun_error = 1UL;
2654 }
2655 else
2656 {
2657 /* Check DMA configuration */
2658 if (tmp_multimode_config != LL_ADC_MULTI_INDEPENDENT)
2659 {
2660 /* Multimode (when feature is available) is enabled,
2661 Common Control Register MDMA bits must be checked. */
2662 if (LL_ADC_GetMultiDMATransfer(__LL_ADC_COMMON_INSTANCE(hadc->Instance)) != LL_ADC_MULTI_REG_DMA_EACH_ADC)
2663 {
2664 overrun_error = 1UL;
2665 }
2666 }
2667 else
2668 {
2669 /* Multimode not set or feature not available or ADC independent */
2670 if ((hadc->Instance->CFGR & ADC_CFGR_DMNGT) != 0UL)
2671 {
2672 overrun_error = 1UL;
2673 }
2674 }
2675 }
2676
2677 if (overrun_error == 1UL)
2678 {
2679 /* Change ADC state to error state */
2680 SET_BIT(hadc->State, HAL_ADC_STATE_REG_OVR);
2681
2682 /* Set ADC error code to overrun */
2683 SET_BIT(hadc->ErrorCode, HAL_ADC_ERROR_OVR);
2684
2685 /* Error callback */
2686 /* Note: In case of overrun, ADC conversion data is preserved until */
2687 /* flag OVR is reset. */
2688 /* Therefore, old ADC conversion data can be retrieved in */
2689 /* function "HAL_ADC_ErrorCallback()". */
2690#if (USE_HAL_ADC_REGISTER_CALLBACKS == 1)
2691 hadc->ErrorCallback(hadc);
2692#else
2693 HAL_ADC_ErrorCallback(hadc);
2694#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */
2695 }
2696
2697 /* Clear ADC overrun flag */
2698 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_OVR);
2699 }
2700
2701 /* ========== Check Injected context queue overflow flag ========== */
2702 if (((tmp_isr & ADC_FLAG_JQOVF) == ADC_FLAG_JQOVF) && ((tmp_ier & ADC_IT_JQOVF) == ADC_IT_JQOVF))
2703 {
2704 /* Change ADC state to overrun state */
2705 SET_BIT(hadc->State, HAL_ADC_STATE_INJ_JQOVF);
2706
2707 /* Set ADC error code to Injected context queue overflow */
2708 SET_BIT(hadc->ErrorCode, HAL_ADC_ERROR_JQOVF);
2709
2710 /* Clear the Injected context queue overflow flag */
2711 __HAL_ADC_CLEAR_FLAG(hadc, ADC_FLAG_JQOVF);
2712
2713 /* Injected context queue overflow callback */
2714#if (USE_HAL_ADC_REGISTER_CALLBACKS == 1)
2715 hadc->InjectedQueueOverflowCallback(hadc);
2716#else
2717 HAL_ADCEx_InjectedQueueOverflowCallback(hadc);
2718#endif /* USE_HAL_ADC_REGISTER_CALLBACKS */
2719 }
2720
2721}
