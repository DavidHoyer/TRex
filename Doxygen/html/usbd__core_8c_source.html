89USBD_StatusTypeDef USBD_Init(USBD_HandleTypeDef *pdev,
90 USBD_DescriptorsTypeDef *pdesc, uint8_t id)
91{
92 USBD_StatusTypeDef ret;
93
94 /* Check whether the USB Host handle is valid */
95 if (pdev == NULL)
96 {
97#if (USBD_DEBUG_LEVEL > 1U)
98 USBD_ErrLog("Invalid Device handle");
99#endif
100 return USBD_FAIL;
101 }
102
103 /* Unlink previous class resources */
104 pdev->pClass = NULL;
105 pdev->pUserData = NULL;
106 pdev->pConfDesc = NULL;
107

113

288USBD_StatusTypeDef USBD_LL_SetupStage(USBD_HandleTypeDef *pdev, uint8_t *psetup)
289{
290 USBD_StatusTypeDef ret;
291
292 USBD_ParseSetupRequest(&pdev->request, psetup);
293
294 pdev->ep0_state = USBD_EP0_SETUP;
295
296 pdev->ep0_data_len = pdev->request.wLength;
297
298 switch (pdev->request.bmRequest & 0x1FU)
299 {
300 case USB_REQ_RECIPIENT_DEVICE:
301 ret = USBD_StdDevReq(pdev, &pdev->request);
302 break;
303
304 case USB_REQ_RECIPIENT_INTERFACE:
305 ret = USBD_StdItfReq(pdev, &pdev->request);
306 break;
307
308 case USB_REQ_RECIPIENT_ENDPOINT:
309 ret = USBD_StdEPReq(pdev, &pdev->request);
310 break;
311
312 default:
313 ret = USBD_LL_StallEP(pdev, (pdev->request.bmRequest & 0x80U));
314 break;
315 }
316
317 return ret;
318}

399USBD_StatusTypeDef USBD_LL_DataInStage(USBD_HandleTypeDef *pdev,
400 uint8_t epnum, uint8_t *pdata)
401{
402 USBD_EndpointTypeDef *pep;
403 USBD_StatusTypeDef ret;
404
405 if (epnum == 0U)
406 {
407 pep = &pdev->ep_in[0];
408
409 if (pdev->ep0_state == USBD_EP0_DATA_IN)
410 {
411 if (pep->rem_length > pep->maxpacket)
412 {
413 pep->rem_length -= pep->maxpacket;
414
415 (void)USBD_CtlContinueSendData(pdev, pdata, pep->rem_length);
416
417 /* Prepare endpoint for premature end of transfer */
418 (void)USBD_LL_PrepareReceive(pdev, 0U, NULL, 0U);
419 }
420 else
421 {
422 /* last packet is MPS multiple, so send ZLP packet */
423 if ((pep->maxpacket == pep->rem_length) &&
424 (pep->total_length >= pep->maxpacket) &&
425 (pep->total_length < pdev->ep0_data_len))
426 {
427 (void)USBD_CtlContinueSendData(pdev, NULL, 0U);
428 pdev->ep0_data_len = 0U;

432 }
433 else

435 if (pdev->dev_state == USBD_STATE_CONFIGURED)
436 {
437 if (pdev->pClass->EP0_TxSent != NULL)
438 {
439 pdev->pClass->EP0_TxSent(pdev);
440 }

490USBD_StatusTypeDef USBD_LL_Reset(USBD_HandleTypeDef *pdev)
491{
492 /* Upon Reset call user call back */
493 pdev->dev_state = USBD_STATE_DEFAULT;
494 pdev->ep0_state = USBD_EP0_IDLE;
495 pdev->dev_config = 0U;
496 pdev->dev_remote_wakeup = 0U;
497
498 if (pdev->pClass == NULL)
499 {
500 return USBD_FAIL;
501 }
502
503 if (pdev->pClassData != NULL)
504 {
505 if (pdev->pClass->DeInit != NULL)
506 {
507 (void)pdev->pClass->DeInit(pdev, (uint8_t)pdev->dev_config);
508 }
509 }
510
511 /* Open EP0 OUT */
512 (void)USBD_LL_OpenEP(pdev, 0x00U, USBD_EP_TYPE_CTRL, USB_MAX_EP0_SIZE);
513 pdev->ep_out[0x00U & 0xFU].is_used = 1U;
514
515 pdev->ep_out[0].maxpacket = USB_MAX_EP0_SIZE;
516
517 /* Open EP0 IN */
518 (void)USBD_LL_OpenEP(pdev, 0x80U, USBD_EP_TYPE_CTRL, USB_MAX_EP0_SIZE);
519 pdev->ep_in[0x80U & 0xFU].is_used = 1U;
520
521 pdev->ep_in[0].maxpacket = USB_MAX_EP0_SIZE;
522
523 return USBD_OK;
524}
