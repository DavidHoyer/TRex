148result_t TC3A227E_GetColor(color_t *value) {
149 while (!IsDone())
150 CORE_Delay(1);
151
152 if (value == NULL)
153 return RESULT_INVALID_ARGUMENTS;
154
155 uint8_t l, h;
156
157 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, RDATA | COMMAND, &l));
158 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, RDATAH | COMMAND, &h));
159
160 value->r = (h << 8) | l;
161
162 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, GDATA | COMMAND, &l));
163 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, GDATAH | COMMAND, &h));
164
165 value->g = (h << 8) | l;
166
167 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, BDATA | COMMAND, &l));
168 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, BDATAH | COMMAND, &h));
169
170 value->b = (h << 8) | l;
171
172 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, CDATA | COMMAND, &l));
173 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, CDATAH | COMMAND, &h));
174
175 value->a = (h << 8) | l;
176
177 R_TRY(StartConversion());
178
179 return RESULT_SUCCESS;
180}
