552 }
553 }
554
555 /* Change DAC state */
556 hdac->State = HAL_DAC_STATE_READY;

607HAL_StatusTypeDef HAL_DAC_Start_DMA(DAC_HandleTypeDef *hdac, uint32_t Channel, uint32_t *pData, uint32_t Length,
608 uint32_t Alignment)
609{
610 HAL_StatusTypeDef status;
611 uint32_t tmpreg = 0U;
612
613 /* Check the parameters */
614 assert_param(IS_DAC_CHANNEL(Channel));
615 assert_param(IS_DAC_ALIGN(Alignment));
616
617 /* Process locked */
618 __HAL_LOCK(hdac);
619
620 /* Change DAC state */
621 hdac->State = HAL_DAC_STATE_BUSY;
622
623 if (Channel == DAC_CHANNEL_1)
624 {
625 /* Set the DMA transfer complete callback for channel1 */
626 hdac->DMA_Handle1->XferCpltCallback = DAC_DMAConvCpltCh1;
627
628 /* Set the DMA half transfer complete callback for channel1 */
629 hdac->DMA_Handle1->XferHalfCpltCallback = DAC_DMAHalfConvCpltCh1;
630
631 /* Set the DMA error callback for channel1 */
632 hdac->DMA_Handle1->XferErrorCallback = DAC_DMAErrorCh1;
633
634 /* Enable the selected DAC channel1 DMA request */
635 SET_BIT(hdac->Instance->CR, DAC_CR_DMAEN1);
636
637 /* Case of use of channel 1 */
638 switch (Alignment)
639 {
640 case DAC_ALIGN_12B_R:
641 /* Get DHR12R1 address */
642 tmpreg = (uint32_t)&hdac->Instance->DHR12R1;
643 break;
644 case DAC_ALIGN_12B_L:
645 /* Get DHR12L1 address */
646 tmpreg = (uint32_t)&hdac->Instance->DHR12L1;
647 break;
648 case DAC_ALIGN_8B_R:
649 /* Get DHR8R1 address */
650 tmpreg = (uint32_t)&hdac->Instance->DHR8R1;
651 break;
652 default:
653 break;
654 }
655 }
656 else
657 {
658 /* Set the DMA transfer complete callback for channel2 */
659 hdac->DMA_Handle2->XferCpltCallback = DAC_DMAConvCpltCh2;
660
661 /* Set the DMA half transfer complete callback for channel2 */
662 hdac->DMA_Handle2->XferHalfCpltCallback = DAC_DMAHalfConvCpltCh2;
663
664 /* Set the DMA error callback for channel2 */
665 hdac->DMA_Handle2->XferErrorCallback = DAC_DMAErrorCh2;
666
667 /* Enable the selected DAC channel2 DMA request */
668 SET_BIT(hdac->Instance->CR, DAC_CR_DMAEN2);
669
670 /* Case of use of channel 2 */
671 switch (Alignment)
672 {
673 case DAC_ALIGN_12B_R:
674 /* Get DHR12R2 address */
675 tmpreg = (uint32_t)&hdac->Instance->DHR12R2;
676 break;
677 case DAC_ALIGN_12B_L:
678 /* Get DHR12L2 address */
679 tmpreg = (uint32_t)&hdac->Instance->DHR12L2;
680 break;
681 case DAC_ALIGN_8B_R:
682 /* Get DHR8R2 address */
683 tmpreg = (uint32_t)&hdac->Instance->DHR8R2;
684 break;
685 default:
686 break;
687 }
688 }
689
690 /* Enable the DMA Stream */
691 if (Channel == DAC_CHANNEL_1)
692 {
693 /* Enable the DAC DMA underrun interrupt */
694 __HAL_DAC_ENABLE_IT(hdac, DAC_IT_DMAUDR1);
695
696 /* Enable the DMA Stream */
697 status = HAL_DMA_Start_IT(hdac->DMA_Handle1, (uint32_t)pData, tmpreg, Length);
698 }
699 else
700 {
701 /* Enable the DAC DMA underrun interrupt */
702 __HAL_DAC_ENABLE_IT(hdac, DAC_IT_DMAUDR2);
703
704 /* Enable the DMA Stream */
705 status = HAL_DMA_Start_IT(hdac->DMA_Handle2, (uint32_t)pData, tmpreg, Length);
706 }
707
708 /* Process Unlocked */
709 __HAL_UNLOCK(hdac);
710
711 if (status == HAL_OK)
712 {
713 /* Enable the Peripheral */
714 __HAL_DAC_ENABLE(hdac, Channel);
715 }
716 else
717 {
718 hdac->ErrorCode |= HAL_DAC_ERROR_DMA;
719 }
720
721 /* Return function status */
722 return status;
723}

781void HAL_DAC_IRQHandler(DAC_HandleTypeDef *hdac)
782{
783 if (__HAL_DAC_GET_IT_SOURCE(hdac, DAC_IT_DMAUDR1))
784 {
785 /* Check underrun flag of DAC channel 1 */
786 if (__HAL_DAC_GET_FLAG(hdac, DAC_FLAG_DMAUDR1))
787 {
788 /* Change DAC state to error state */
789 hdac->State = HAL_DAC_STATE_ERROR;
790
791 /* Set DAC error code to chanel1 DMA underrun error */
792 SET_BIT(hdac->ErrorCode, HAL_DAC_ERROR_DMAUNDERRUNCH1);
793
794 /* Clear the underrun flag */
795 __HAL_DAC_CLEAR_FLAG(hdac, DAC_FLAG_DMAUDR1);
796
797 /* Disable the selected DAC channel1 DMA request */
798 CLEAR_BIT(hdac->Instance->CR, DAC_CR_DMAEN1);
799
800 /* Error callback */
801#if (USE_HAL_DAC_REGISTER_CALLBACKS == 1)
802 hdac->DMAUnderrunCallbackCh1(hdac);
803#else
804 HAL_DAC_DMAUnderrunCallbackCh1(hdac);
805#endif /* USE_HAL_DAC_REGISTER_CALLBACKS */
806 }
807 }
808
809 if (__HAL_DAC_GET_IT_SOURCE(hdac, DAC_IT_DMAUDR2))
810 {
811 /* Check underrun flag of DAC channel 2 */
812 if (__HAL_DAC_GET_FLAG(hdac, DAC_FLAG_DMAUDR2))
813 {
814 /* Change DAC state to error state */
815 hdac->State = HAL_DAC_STATE_ERROR;
816
817 /* Set DAC error code to channel2 DMA underrun error */
818 SET_BIT(hdac->ErrorCode, HAL_DAC_ERROR_DMAUNDERRUNCH2);
819
820 /* Clear the underrun flag */
821 __HAL_DAC_CLEAR_FLAG(hdac, DAC_FLAG_DMAUDR2);
822
823 /* Disable the selected DAC channel2 DMA request */
824 CLEAR_BIT(hdac->Instance->CR, DAC_CR_DMAEN2);
825
826 /* Error callback */
827#if (USE_HAL_DAC_REGISTER_CALLBACKS == 1)
828 hdac->DMAUnderrunCallbackCh2(hdac);
829#else
830 HAL_DACEx_DMAUnderrunCallbackCh2(hdac);
831#endif /* USE_HAL_DAC_REGISTER_CALLBACKS */

998HAL_StatusTypeDef HAL_DAC_ConfigChannel(DAC_HandleTypeDef *hdac, DAC_ChannelConfTypeDef *sConfig, uint32_t Channel)
999{
1000 uint32_t tmpreg1;
1001 uint32_t tmpreg2;
1002 uint32_t tickstart;
1003 uint32_t connectOnChip;
1004
1005 /* Check the DAC parameters */
1006 assert_param(IS_DAC_TRIGGER(sConfig->DAC_Trigger));
1007 assert_param(IS_DAC_OUTPUT_BUFFER_STATE(sConfig->DAC_OutputBuffer));
1008 assert_param(IS_DAC_CHIP_CONNECTION(sConfig->DAC_ConnectOnChipPeripheral));
1009 assert_param(IS_DAC_TRIMMING(sConfig->DAC_UserTrimming));
1010 if ((sConfig->DAC_UserTrimming) == DAC_TRIMMING_USER)
1011 {
1012 assert_param(IS_DAC_TRIMMINGVALUE(sConfig->DAC_TrimmingValue));
1013 }
1014 assert_param(IS_DAC_SAMPLEANDHOLD(sConfig->DAC_SampleAndHold));
1015 if ((sConfig->DAC_SampleAndHold) == DAC_SAMPLEANDHOLD_ENABLE)
1016 {
1017 assert_param(IS_DAC_SAMPLETIME(sConfig->DAC_SampleAndHoldConfig.DAC_SampleTime));
1018 assert_param(IS_DAC_HOLDTIME(sConfig->DAC_SampleAndHoldConfig.DAC_HoldTime));
1019 assert_param(IS_DAC_REFRESHTIME(sConfig->DAC_SampleAndHoldConfig.DAC_RefreshTime));
1020 }
1021 assert_param(IS_DAC_CHANNEL(Channel));
1022
1023 /* Process locked */
1024 __HAL_LOCK(hdac);
1025
1026 /* Change DAC state */
1027 hdac->State = HAL_DAC_STATE_BUSY;
1028
1029 /* Sample and hold configuration */
1030 if (sConfig->DAC_SampleAndHold == DAC_SAMPLEANDHOLD_ENABLE)
1031 {
1032 /* Get timeout */
1033 tickstart = HAL_GetTick();
1034
1035 if (Channel == DAC_CHANNEL_1)
1036 {
1037 /* SHSR1 can be written when BWST1 is cleared */
1038 while (((hdac->Instance->SR) & DAC_SR_BWST1) != 0UL)
1039 {
1040 /* Check for the Timeout */
1041 if ((HAL_GetTick() - tickstart) > TIMEOUT_DAC_CALIBCONFIG)
1042 {
1043 /* Update error code */
1044 SET_BIT(hdac->ErrorCode, HAL_DAC_ERROR_TIMEOUT);
1045
1046 /* Change the DMA state */
1047 hdac->State = HAL_DAC_STATE_TIMEOUT;
1048
1049 return HAL_TIMEOUT;
1050 }
1051 }
1052 HAL_Delay(1);
1053 hdac->Instance->SHSR1 = sConfig->DAC_SampleAndHoldConfig.DAC_SampleTime;
1054 }
1055 else /* Channel 2 */
1056 {
1057 /* SHSR2 can be written when BWST2 is cleared */
1058 while (((hdac->Instance->SR) & DAC_SR_BWST2) != 0UL)
1059 {
1060 /* Check for the Timeout */
1061 if ((HAL_GetTick() - tickstart) > TIMEOUT_DAC_CALIBCONFIG)
1062 {
1063 /* Update error code */
1064 SET_BIT(hdac->ErrorCode, HAL_DAC_ERROR_TIMEOUT);
1065
1066 /* Change the DMA state */
1067 hdac->State = HAL_DAC_STATE_TIMEOUT;
1068
1069 return HAL_TIMEOUT;
1070 }
1071 }
1072 HAL_Delay(1U);
1073 hdac->Instance->SHSR2 = sConfig->DAC_SampleAndHoldConfig.DAC_SampleTime;
1074 }
1075
1076 /* HoldTime */
1077 MODIFY_REG(hdac->Instance->SHHR, DAC_SHHR_THOLD1 << (Channel & 0x10UL),
1078 (sConfig->DAC_SampleAndHoldConfig.DAC_HoldTime) << (Channel & 0x10UL));
1079 /* RefreshTime */
1080 MODIFY_REG(hdac->Instance->SHRR, DAC_SHRR_TREFRESH1 << (Channel & 0x10UL),
1081 (sConfig->DAC_SampleAndHoldConfig.DAC_RefreshTime) << (Channel & 0x10UL));
1082 }
1083
1084 if (sConfig->DAC_UserTrimming == DAC_TRIMMING_USER)
1085 /* USER TRIMMING */
1086 {
1087 /* Get the DAC CCR value */
1088 tmpreg1 = hdac->Instance->CCR;
1089 /* Clear trimming value */
1090 tmpreg1 &= ~(((uint32_t)(DAC_CCR_OTRIM1)) << (Channel & 0x10UL));
1091 /* Configure for the selected trimming offset */
1092 tmpreg2 = sConfig->DAC_TrimmingValue;
1093 /* Calculate CCR register value depending on DAC_Channel */
1094 tmpreg1 |= tmpreg2 << (Channel & 0x10UL);
1095 /* Write to DAC CCR */
1096 hdac->Instance->CCR = tmpreg1;
1097 }
1098 /* else factory trimming is used (factory setting are available at reset)*/
1099 /* SW Nothing has nothing to do */
1100
1101 /* Get the DAC MCR value */
1102 tmpreg1 = hdac->Instance->MCR;
1103 /* Clear DAC_MCR_MODEx bits */
1104 tmpreg1 &= ~(((uint32_t)(DAC_MCR_MODE1)) << (Channel & 0x10UL));
1105 /* Configure for the selected DAC channel: mode, buffer output & on chip peripheral connect */
1106 if (sConfig->DAC_ConnectOnChipPeripheral == DAC_CHIPCONNECT_EXTERNAL)
1107 {
1108 connectOnChip = 0x00000000UL;
1109 }
1110 else if (sConfig->DAC_ConnectOnChipPeripheral == DAC_CHIPCONNECT_INTERNAL)
1111 {
1112 connectOnChip = DAC_MCR_MODE1_0;
1113 }
1114 else /* (sConfig->DAC_ConnectOnChipPeripheral == DAC_CHIPCONNECT_BOTH) */
1115 {
1116 if (sConfig->DAC_OutputBuffer == DAC_OUTPUTBUFFER_ENABLE)
1117 {
1118 connectOnChip = DAC_MCR_MODE1_0;
1119 }
1120 else
1121 {
1122 connectOnChip = 0x00000000UL;
1123 }
1124 }
1125 tmpreg2 = (sConfig->DAC_SampleAndHold | sConfig->DAC_OutputBuffer | connectOnChip);
1126 /* Calculate MCR register value depending on DAC_Channel */
1127 tmpreg1 |= tmpreg2 << (Channel & 0x10UL);
1128 /* Write to DAC MCR */
1129 hdac->Instance->MCR = tmpreg1;
1130
1131 /* DAC in normal operating mode hence clear DAC_CR_CENx bit */
1132 CLEAR_BIT(hdac->Instance->CR, DAC_CR_CEN1 << (Channel & 0x10UL));
1133
1134 /* Get the DAC CR value */
1135 tmpreg1 = hdac->Instance->CR;
1136 /* Clear TENx, TSELx, WAVEx and MAMPx bits */
1137 tmpreg1 &= ~(((uint32_t)(DAC_CR_MAMP1 | DAC_CR_WAVE1 | DAC_CR_TSEL1 | DAC_CR_TEN1)) << (Channel & 0x10UL));
1138 /* Configure for the selected DAC channel: trigger */
1139 /* Set TSELx and TENx bits according to DAC_Trigger value */
1140 tmpreg2 = sConfig->DAC_Trigger;
1141 /* Calculate CR register value depending on DAC_Channel */
1142 tmpreg1 |= tmpreg2 << (Channel & 0x10UL);
1143 /* Write to DAC CR */
1144 hdac->Instance->CR = tmpreg1;
1145 /* Disable wave generation */
1146 hdac->Instance->CR &= ~(DAC_CR_WAVE1 << (Channel & 0x10UL));
1147
1148 /* Change DAC state */
1149 hdac->State = HAL_DAC_STATE_READY;
1150
1151 /* Process unlocked */
1152 __HAL_UNLOCK(hdac);
1153
1154 /* Return function status */
1155 return HAL_OK;
1156}
