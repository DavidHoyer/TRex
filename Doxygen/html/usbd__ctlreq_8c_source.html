214USBD_StatusTypeDef USBD_StdEPReq(USBD_HandleTypeDef *pdev, USBD_SetupReqTypedef *req)
215{
216 USBD_EndpointTypeDef *pep;
217 uint8_t ep_addr;
218 USBD_StatusTypeDef ret = USBD_OK;
219 ep_addr = LOBYTE(req->wIndex);
220
221 switch (req->bmRequest & USB_REQ_TYPE_MASK)
222 {
223 case USB_REQ_TYPE_CLASS:
224 case USB_REQ_TYPE_VENDOR:
225 ret = (USBD_StatusTypeDef)pdev->pClass->Setup(pdev, req);
226 break;
227
228 case USB_REQ_TYPE_STANDARD:
229 switch (req->bRequest)
230 {
231 case USB_REQ_SET_FEATURE:
232 switch (pdev->dev_state)
233 {
234 case USBD_STATE_ADDRESSED:
235 if ((ep_addr != 0x00U) && (ep_addr != 0x80U))
236 {
237 (void)USBD_LL_StallEP(pdev, ep_addr);
238 (void)USBD_LL_StallEP(pdev, 0x80U);
239 }
240 else
241 {
242 USBD_CtlError(pdev, req);
243 }
244 break;
245
246 case USBD_STATE_CONFIGURED:
247 if (req->wValue == USB_FEATURE_EP_HALT)
248 {
249 if ((ep_addr != 0x00U) && (ep_addr != 0x80U) && (req->wLength == 0x00U))
250 {
251 (void)USBD_LL_StallEP(pdev, ep_addr);
252 }
253 }
254 (void)USBD_CtlSendStatus(pdev);
255
256 break;
257
258 default:
259 USBD_CtlError(pdev, req);
260 break;
261 }
262 break;
263
264 case USB_REQ_CLEAR_FEATURE:
265
266 switch (pdev->dev_state)
267 {
268 case USBD_STATE_ADDRESSED:
269 if ((ep_addr != 0x00U) && (ep_addr != 0x80U))
270 {
271 (void)USBD_LL_StallEP(pdev, ep_addr);
272 (void)USBD_LL_StallEP(pdev, 0x80U);
273 }
274 else
275 {
276 USBD_CtlError(pdev, req);
277 }
278 break;
279
280 case USBD_STATE_CONFIGURED:
281 if (req->wValue == USB_FEATURE_EP_HALT)
282 {
283 if ((ep_addr & 0x7FU) != 0x00U)
284 {
285 (void)USBD_LL_ClearStallEP(pdev, ep_addr);
286 }
287 (void)USBD_CtlSendStatus(pdev);
288 ret = (USBD_StatusTypeDef)pdev->pClass->Setup(pdev, req);
289 }
290 break;
291
292 default:
293 USBD_CtlError(pdev, req);
294 break;
295 }
296 break;
297
298 case USB_REQ_GET_STATUS:
299 switch (pdev->dev_state)
300 {
301 case USBD_STATE_ADDRESSED:
302 if ((ep_addr != 0x00U) && (ep_addr != 0x80U))
303 {
304 USBD_CtlError(pdev, req);
305 break;
306 }
307 pep = ((ep_addr & 0x80U) == 0x80U) ? &pdev->ep_in[ep_addr & 0x7FU] : \
308 &pdev->ep_out[ep_addr & 0x7FU];
309
310 pep->status = 0x0000U;
311
312 (void)USBD_CtlSendData(pdev, (uint8_t *)&pep->status, 2U);
313 break;
314
315 case USBD_STATE_CONFIGURED:
316 if ((ep_addr & 0x80U) == 0x80U)
317 {
318 if (pdev->ep_in[ep_addr & 0xFU].is_used == 0U)
319 {
320 USBD_CtlError(pdev, req);
321 break;
322 }
323 }
324 else
325 {
326 if (pdev->ep_out[ep_addr & 0xFU].is_used == 0U)
327 {
328 USBD_CtlError(pdev, req);
329 break;
330 }
331 }
332
333 pep = ((ep_addr & 0x80U) == 0x80U) ? &pdev->ep_in[ep_addr & 0x7FU] : \
334 &pdev->ep_out[ep_addr & 0x7FU];
335
336 if ((ep_addr == 0x00U) || (ep_addr == 0x80U))
337 {
338 pep->status = 0x0000U;
339 }
340 else if (USBD_LL_IsStallEP(pdev, ep_addr) != 0U)
341 {
342 pep->status = 0x0001U;
343 }
344 else
345 {
346 pep->status = 0x0000U;
347 }
348
349 (void)USBD_CtlSendData(pdev, (uint8_t *)&pep->status, 2U);
350 break;
351
352 default:
353 USBD_CtlError(pdev, req);
354 break;
355 }
356 break;
357
358 default:
359 USBD_CtlError(pdev, req);
360 break;
361 }
362 break;
363
364 default:
365 USBD_CtlError(pdev, req);
366 break;
367 }

369 return ret;

629static USBD_StatusTypeDef USBD_SetConfig(USBD_HandleTypeDef *pdev, USBD_SetupReqTypedef *req)
630{
631 USBD_StatusTypeDef ret = USBD_OK;
632 static uint8_t cfgidx;
633
634 cfgidx = (uint8_t)(req->wValue);
635
636 if (cfgidx > USBD_MAX_NUM_CONFIGURATION)
637 {
638 USBD_CtlError(pdev, req);
639 return USBD_FAIL;
640 }
641
642 switch (pdev->dev_state)
643 {
644 case USBD_STATE_ADDRESSED:
645 if (cfgidx != 0U)
646 {
647 pdev->dev_config = cfgidx;
648
649 ret = USBD_SetClassConfig(pdev, cfgidx);
650
651 if (ret != USBD_OK)
652 {
653 USBD_CtlError(pdev, req);
654 }
655 else
656 {
657 (void)USBD_CtlSendStatus(pdev);
658 pdev->dev_state = USBD_STATE_CONFIGURED;
659 }
660 }
661 else
662 {
663 (void)USBD_CtlSendStatus(pdev);
664 }
665 break;
666
667 case USBD_STATE_CONFIGURED:
668 if (cfgidx == 0U)
669 {
670 pdev->dev_state = USBD_STATE_ADDRESSED;
671 pdev->dev_config = cfgidx;
672 (void)USBD_ClrClassConfig(pdev, cfgidx);
673 (void)USBD_CtlSendStatus(pdev);
674 }
675 else if (cfgidx != pdev->dev_config)
676 {
677 /* Clear old configuration */
678 (void)USBD_ClrClassConfig(pdev, (uint8_t)pdev->dev_config);
679
680 /* set new configuration */
681 pdev->dev_config = cfgidx;
682
683 ret = USBD_SetClassConfig(pdev, cfgidx);
684
685 if (ret != USBD_OK)
686 {
687 USBD_CtlError(pdev, req);
688 (void)USBD_ClrClassConfig(pdev, (uint8_t)pdev->dev_config);
689 pdev->dev_state = USBD_STATE_ADDRESSED;
690 }
691 else
692 {
693 (void)USBD_CtlSendStatus(pdev);
694 }
695 }
696 else
697 {
698 (void)USBD_CtlSendStatus(pdev);
699 }
700 break;
701
702 default:
703 USBD_CtlError(pdev, req);
704 (void)USBD_ClrClassConfig(pdev, cfgidx);
705 ret = USBD_FAIL;
706 break;
707 }
708
709 return ret;
710}

886{
887 uint8_t idx = 0U;

889
