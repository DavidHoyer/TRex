50result_t SERIAL_Read(char *buffer, size_t size, uint32_t timeout) {
51 if (buffer == NULL || size == 0)
52 return RESULT_SERIAL_READ_ERROR;
53
54 if (HAL_UART_Receive(&huart1, (uint8_t*)buffer, size - 1, timeout) != HAL_OK)
55 return RESULT_SERIAL_READ_ERROR;
56
57 /* Null terminate string in buffer */
58 buffer[size - 1] = '\x00';
59
60 return RESULT_SUCCESS;
61}

63result_t SERIAL_Write(const char *string, uint32_t timeout) {
64 if (string == NULL)
65 return RESULT_SERIAL_WRITE_ERROR;
66
67 if (HAL_UART_Transmit(&huart1, (uint8_t*)string, strlen(string), timeout) != HAL_OK)
68 return RESULT_SERIAL_WRITE_ERROR;
69
70 return RESULT_SUCCESS;
