75result_t LPS22BH_Init(void) {
76 uint8_t readDeviceId = 0x00;
77 do {
78 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, WHO_AM_I | REQUEST_RESPONSE, &readDeviceId));
79 if (readDeviceId == DEVICE_ID)
80 break;
81
82 CORE_Delay(100);
83 } while (true);
84
85 return RESULT_SUCCESS;
86}

89static result_t StartConversion(void) {
90 ControlRegister2_t reg = { .ONE_SHOT = 1 };
91
92 R_TRY(I2C_WriteRegister(DEVICE_ADDRESS, CTRL_REG2, *(uint8_t*)&reg));
93
94 uint8_t conversionDone = 0x00;
95 do {
96 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, CTRL_REG2 | REQUEST_RESPONSE, &conversionDone));
97 if (conversionDone == 0x00)
98 break;
99
100 CORE_Delay(10);
101 } while (true);
102
103 return RESULT_SUCCESS;
104}

107result_t LPS22BH_GetTemperature(float32_t *value) {
108 R_TRY(StartConversion());
109
110 if (value == NULL)
111 return RESULT_INVALID_ARGUMENTS;
112
113 uint8_t l, h;
114
115 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, TEMP_OUT_L, &l));
116 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, TEMP_OUT_H, &h));
117
118 *value = (float32_t)(l | h << 8) / 100.0F;
119
120 return RESULT_SUCCESS;
121}

123result_t LPS22BH_GetPressure(float32_t *value) {
124 R_TRY(StartConversion());
125
126 if (value == NULL)
127 return RESULT_INVALID_ARGUMENTS;
128
129 uint8_t xl, l, h;
130 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, PRESS_OUT_XL, &xl));
131 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, PRESS_OUT_L, &l));
132 R_TRY(I2C_ReadRegister(DEVICE_ADDRESS, PRESS_OUT_H, &h));
133
134 *value = (float32_t)(xl | (l << 8) | (h << 16)) / 4096.0F;
135
136 return RESULT_SUCCESS;
137}
