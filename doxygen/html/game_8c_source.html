244event_T CheckEvent(void)
245{
246 //--- create event variable and touch input variable
247 event_T event;
248 LCD_TouchPosition_t touchPosition = {0, 0};
249 static char flag = FALSE;
250
251 //--- check for touch input
252 if (R_SUCCESS(LCD_TouchGetPosition(&touchPosition)))
253 {
254 //--- if no touch input
255 if(touchPosition.x == 0 && touchPosition.y == 480){
256 //--- set event Flage to FALSE and return event
257 event.eventFlag = FALSE;
258 flag = FALSE;
259 return event;
260 }
261 //--- if flag TRUE from last event
262 if(flag == TRUE){
263 //--- set event Flag to FALSE and return event
264 event.eventFlag = FALSE;
265 return event;
266 }
267 //--- if touch input happend
268 event.eventFlag = TRUE; //--- set event Flag
269 event.x = touchPosition.x; //--- set event position x
270 event.y = touchPosition.y; //--- set event position y
271 flag = TRUE; //--- set Flag for next check
272 }
273
274 return event; //--- return incoming touch event
275}

325 if(obstacleBmp[i].visible == TRUE){

327 if(obstacleBmp[i].x <= 3){
328 //--- if it reached left side, delete

331 else{
332 //--- if not reached left side, shift obstacle
333 ShiftBmp(&obstacleBmp[i], -4, 0);
334 }
335 }
336 else{
337 //--- Obstacle is not visible
338 uint16_t j = i == 0 ? OBSTACLES_NUMBER - 1 : i - 1;
339 //--- if previous obstacle is visible
340 if(obstacleBmp[j].visible == TRUE){
341 //--- give random funktion a seed
342 srand(j*20000/(HAL_GetTick()%1000+ 1));
343 //--- if last obstacle moved 300, 375 or 450 pixels (random distance)
344 if(obstacleBmp[j].x < LCD_WIDTH - (400 + ((rand()%3-1) *75))){
345 //--- create new obstacle
346 DrawBmp(&obstacleBmp[i], LCD_WIDTH - obstacleBmp[i].w, LCD_HEIGHT - GROUND_HEIGHT - obstacleBmp[i].h);
347 }
348 }
349 }
350 }
351 //--- reset timer
352 tickOld = tickNew;

424//*********************************************************************
425//*** Given three collinear points p, q, r, the function checks if ***
426//*** point q lies on line segment 'pr' ***
427//*********************************************************************
428static bool onSegment(pixel_t p, pixel_t q, pixel_t r)
429{
430 if (q.x <= fmax(p.x, r.x) && q.x >= fmin(p.x, r.x) &&
431 q.y <= fmax(p.y, r.y) && q.y >= fmin(p.y, r.y))
432 return true;
433
434 return false;
435}

437//*****************************************************************************
438//*** To find orientation of ordered triplet (p, q, r) ***
439//*** The function returns following values ***
440//*** 0 --> p, q and r are collinear ***
441//*** 1 --> Clockwise ***
442//*** 2 --> Counterclockwise ***
443//*** See https://www.geeksforgeeks.org/orientation-3-ordered-points/ ***
444//*****************************************************************************
445static int orientation(pixel_t p, pixel_t q, pixel_t r)
446{
447 int val = (q.y - p.y) * (r.x - q.x) -
448 (q.x - p.x) * (r.y - q.y);
449
450 if (val == 0) return 0; //--- collinear
451
452 return (val > 0)? 1: 2; //--- clock or counterclock wise
453}

455//*********************************************************************
456//*** Checks if two line segments intersect ***
457//*********************************************************************
458static int doIntersect( uint16_t A1_x, uint16_t A1_y, //Line 1 Pos A
459 uint16_t A2_x, uint16_t A2_y, //Line 1 Pos B
460 uint16_t B1_x, uint16_t B1_y, //Line 2 Pos A
461 uint16_t B2_x, uint16_t B2_y) //Line 2 Pos B
462{
463 pixel_t p1 = {A1_x, A1_y};
464 pixel_t q1 = {A2_x, A2_y};
465 pixel_t p2 = {B1_x, B1_y};
466 pixel_t q2 = {B2_x, B2_y};
467
468 //--- Find the four orientations needed for general and special cases
469 int o1 = orientation(p1, q1, p2);
470 int o2 = orientation(p1, q1, q2);
471 int o3 = orientation(p2, q2, p1);
472 int o4 = orientation(p2, q2, q1);
473
474 //--- General case
475 if (o1 != o2 && o3 != o4)
476 return true;
477
478 //--- Special Cases
479 //--- p1, q1 and p2 are collinear and p2 lies on segment p1q1
480 if (o1 == 0 && onSegment(p1, p2, q1))
481 return true;
482
483 //--- p1, q1 and q2 are collinear and q2 lies on segment p1q1
484 if (o2 == 0 && onSegment(p1, q2, q1))
485 return true;
486
487 //--- p2, q2 and p1 are collinear and p1 lies on segment p2q2
488 if (o3 == 0 && onSegment(p2, p1, q2))
489 return true;
490
491 //--- p2, q2 and q1 are collinear and q1 lies on segment p2q2
492 if (o4 == 0 && onSegment(p2, q1, q2))
493 return true;
494
495 return false; //--- Doesn't fall in any of the above cases
496}

498//*********************************************************************
499//*** Checks if the T-Rex is colliding with one of the obstacles ***
500//*********************************************************************
501char CheckCollision(void){
502
503 //--- if T-Rex is not visible, break
504 if(tRexBmp.visible == FALSE)
505 return FALSE;
506
507 //--- loop through every obstacle
508 for(uint16_t i = 0; i < OBSTACLES_NUMBER; i++){
509 //--- if obstacle is not visible, skip this obstacle
510 if(obstacleBmp[i].visible == FALSE)
511 continue;
512
513 //--- Check if the image borders of the T-Rex and the obstacles are colliding
514 if (tRexBmp.x + tRexBmp.w < obstacleBmp[i].x ||
515 tRexBmp.x > obstacleBmp[i].x + obstacleBmp[i].w ||
516 tRexBmp.y + tRexBmp.h < obstacleBmp[i].y ||
517 tRexBmp.y > obstacleBmp[i].y + obstacleBmp[i].h)
518 {
519 continue;
520 }
521
522 pixel_t *TRexPixelA = tRexBmp.head; //--- Point 1 of T-Rex Border
523 pixel_t *TRexPixelB = TRexPixelA->next; //--- Point 2 of T-Rex Border
524 pixel_t *ObstPixelA = obstacleBmp[0].head; //--- Point 1 of Obstacle Border
525 pixel_t *ObstPixelB = ObstPixelA->next; //--- Point 2 of Obstacle Border
526
527 while(ObstPixelA != NULL && ObstPixelB != NULL){
528 while(TRexPixelA != NULL && TRexPixelB != NULL){
529 //--- check for a collision between Lines
530 if(doIntersect( obstacleBmp[i].x + ObstPixelA->x, obstacleBmp[i].y + ObstPixelA->y, //--- Line 1 Pos A
531 obstacleBmp[i].x + ObstPixelB->x, obstacleBmp[i].y + ObstPixelB->y, //--- Line 1 Pos B
532 tRexBmp.x + TRexPixelA->x, tRexBmp.y + TRexPixelA->y, //--- Line 2 Pos A
533 tRexBmp.x + TRexPixelB->x, tRexBmp.y + TRexPixelB->y //--- Line 2 Pos B
534 ) == true)
535 {
536 return true;
537 }
538
539 //--- Next line of the T-Rex border
540 TRexPixelA = TRexPixelB;
541 TRexPixelB = TRexPixelA->next;
542 }
543
544 //--- Next line of the Obstacle border
545 ObstPixelA = ObstPixelB;
546 ObstPixelB = ObstPixelA->next;
547
548 //--- Start again on T-Rex border
549 TRexPixelA = tRexBmp.head;
550 TRexPixelB = TRexPixelA->next;
551 }
552 }
553 return false;
554}
