354static uint8_t USBD_CUSTOM_HID_Init(USBD_HandleTypeDef *pdev, uint8_t cfgidx)
355{
356 UNUSED(cfgidx);
357 USBD_CUSTOM_HID_HandleTypeDef *hhid;
358
359 hhid = USBD_malloc(sizeof(USBD_CUSTOM_HID_HandleTypeDef));
360
361 if (hhid == NULL)
362 {
363 pdev->pClassData = NULL;
364 return (uint8_t)USBD_EMEM;
365 }
366
367 pdev->pClassData = (void *)hhid;
368
369 if (pdev->dev_speed == USBD_SPEED_HIGH)
370 {
371 pdev->ep_in[CUSTOM_HID_EPIN_ADDR & 0xFU].bInterval = CUSTOM_HID_HS_BINTERVAL;
372 pdev->ep_out[CUSTOM_HID_EPOUT_ADDR & 0xFU].bInterval = CUSTOM_HID_HS_BINTERVAL;
373 }
374 else /* LOW and FULL-speed endpoints */
375 {
376 pdev->ep_in[CUSTOM_HID_EPIN_ADDR & 0xFU].bInterval = CUSTOM_HID_FS_BINTERVAL;
377 pdev->ep_out[CUSTOM_HID_EPOUT_ADDR & 0xFU].bInterval = CUSTOM_HID_FS_BINTERVAL;
378 }
379
380 /* Open EP IN */
381 (void)USBD_LL_OpenEP(pdev, CUSTOM_HID_EPIN_ADDR, USBD_EP_TYPE_INTR,
382 CUSTOM_HID_EPIN_SIZE);
383
384 pdev->ep_in[CUSTOM_HID_EPIN_ADDR & 0xFU].is_used = 1U;
385
386 /* Open EP OUT */
387 (void)USBD_LL_OpenEP(pdev, CUSTOM_HID_EPOUT_ADDR, USBD_EP_TYPE_INTR,
388 CUSTOM_HID_EPOUT_SIZE);
389
390 pdev->ep_out[CUSTOM_HID_EPOUT_ADDR & 0xFU].is_used = 1U;
391
392 hhid->state = CUSTOM_HID_IDLE;
393
394 ((USBD_CUSTOM_HID_ItfTypeDef *)pdev->pUserData)->Init();
395
396 /* Prepare Out endpoint to receive 1st packet */
397 (void)USBD_LL_PrepareReceive(pdev, CUSTOM_HID_EPOUT_ADDR, hhid->Report_buf,
398 USBD_CUSTOMHID_OUTREPORT_BUF_SIZE);
399
400 return (uint8_t)USBD_OK;
401}

429 pdev->pClassData = NULL;
430 }
431
432 return (uint8_t)USBD_OK;

442static uint8_t USBD_CUSTOM_HID_Setup(USBD_HandleTypeDef *pdev,
443 USBD_SetupReqTypedef *req)
444{
445 USBD_CUSTOM_HID_HandleTypeDef *hhid = (USBD_CUSTOM_HID_HandleTypeDef *)pdev->pClassData;
446 uint16_t len = 0U;
447 uint8_t *pbuf = NULL;
448 uint16_t status_info = 0U;
449 USBD_StatusTypeDef ret = USBD_OK;
450
451 if (hhid == NULL)
452 {
453 return (uint8_t)USBD_FAIL;
454 }
455
456 switch (req->bmRequest & USB_REQ_TYPE_MASK)
457 {
458 case USB_REQ_TYPE_CLASS:
459 switch (req->bRequest)
460 {
461 case CUSTOM_HID_REQ_SET_PROTOCOL:
462 hhid->Protocol = (uint8_t)(req->wValue);
463 break;
464
465 case CUSTOM_HID_REQ_GET_PROTOCOL:
466 (void)USBD_CtlSendData(pdev, (uint8_t *)&hhid->Protocol, 1U);
467 break;
468
469 case CUSTOM_HID_REQ_SET_IDLE:
470 hhid->IdleState = (uint8_t)(req->wValue >> 8);
471 break;
472
473 case CUSTOM_HID_REQ_GET_IDLE:
474 (void)USBD_CtlSendData(pdev, (uint8_t *)&hhid->IdleState, 1U);
475 break;
476
477 case CUSTOM_HID_REQ_SET_REPORT:
478 hhid->IsReportAvailable = 1U;
479 (void)USBD_CtlPrepareRx(pdev, hhid->Report_buf, req->wLength);
480 break;
481
482 default:
483 USBD_CtlError(pdev, req);
484 ret = USBD_FAIL;
485 break;
486 }
487 break;
488
489 case USB_REQ_TYPE_STANDARD:
490 switch (req->bRequest)
491 {
492 case USB_REQ_GET_STATUS:
493 if (pdev->dev_state == USBD_STATE_CONFIGURED)
494 {
495 (void)USBD_CtlSendData(pdev, (uint8_t *)&status_info, 2U);
496 }
497 else
498 {
499 USBD_CtlError(pdev, req);
500 ret = USBD_FAIL;
501 }
502 break;
503
504 case USB_REQ_GET_DESCRIPTOR:
505 if ((req->wValue >> 8) == CUSTOM_HID_REPORT_DESC)
506 {
507 len = MIN(USBD_CUSTOM_HID_REPORT_DESC_SIZE, req->wLength);
508 pbuf = ((USBD_CUSTOM_HID_ItfTypeDef *)pdev->pUserData)->pReport;
509 }
510 else
511 {
512 if ((req->wValue >> 8) == CUSTOM_HID_DESCRIPTOR_TYPE)
513 {
514 pbuf = USBD_CUSTOM_HID_Desc;
515 len = MIN(USB_CUSTOM_HID_DESC_SIZ, req->wLength);
516 }
517 }
518
519 (void)USBD_CtlSendData(pdev, pbuf, len);
520 break;
521
522 case USB_REQ_GET_INTERFACE:
523 if (pdev->dev_state == USBD_STATE_CONFIGURED)
524 {
525 (void)USBD_CtlSendData(pdev, (uint8_t *)&hhid->AltSetting, 1U);
526 }
527 else
528 {
529 USBD_CtlError(pdev, req);
530 ret = USBD_FAIL;
531 }
532 break;
533
534 case USB_REQ_SET_INTERFACE:
535 if (pdev->dev_state == USBD_STATE_CONFIGURED)
536 {
537 hhid->AltSetting = (uint8_t)(req->wValue);
538 }
539 else
540 {
541 USBD_CtlError(pdev, req);
542 ret = USBD_FAIL;
543 }
544 break;
545
546 case USB_REQ_CLEAR_FEATURE:
547 break;
548
549 default:
550 USBD_CtlError(pdev, req);
551 ret = USBD_FAIL;
552 break;
553 }
554 break;
555
556 default:
557 USBD_CtlError(pdev, req);
558 ret = USBD_FAIL;
559 break;
560 }
561 return (uint8_t)ret;
562}

571uint8_t USBD_CUSTOM_HID_SendReport(USBD_HandleTypeDef *pdev,
572 uint8_t *report, uint16_t len)
573{
574 USBD_CUSTOM_HID_HandleTypeDef *hhid;
575
576 if (pdev->pClassData == NULL)
577 {
578 return (uint8_t)USBD_FAIL;
579 }
580
581 hhid = (USBD_CUSTOM_HID_HandleTypeDef *)pdev->pClassData;
582
583 if (pdev->dev_state == USBD_STATE_CONFIGURED)
584 {
585 if (hhid->state == CUSTOM_HID_IDLE)
586 {
587 hhid->state = CUSTOM_HID_BUSY;
588 (void)USBD_LL_Transmit(pdev, CUSTOM_HID_EPIN_ADDR, report, len);
589 }
590 else
591 {
592 return (uint8_t)USBD_BUSY;
593 }
594 }
595 return (uint8_t)USBD_OK;
596}
