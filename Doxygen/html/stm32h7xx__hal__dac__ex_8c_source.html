115HAL_StatusTypeDef HAL_DACEx_DualStart(DAC_HandleTypeDef *hdac)
116{
117 uint32_t tmp_swtrig = 0UL;
118
119
120 /* Process locked */
121 __HAL_LOCK(hdac);
122
123 /* Change DAC state */
124 hdac->State = HAL_DAC_STATE_BUSY;
125
126 /* Enable the Peripheral */
127 __HAL_DAC_ENABLE(hdac, DAC_CHANNEL_1);
128 __HAL_DAC_ENABLE(hdac, DAC_CHANNEL_2);
129
130 /* Check if software trigger enabled */
131 if ((hdac->Instance->CR & (DAC_CR_TEN1 | DAC_CR_TSEL1)) == DAC_TRIGGER_SOFTWARE)
132 {
133 tmp_swtrig |= DAC_SWTRIGR_SWTRIG1;
134 }
135 if ((hdac->Instance->CR & (DAC_CR_TEN2 | DAC_CR_TSEL2)) == (DAC_TRIGGER_SOFTWARE << (DAC_CHANNEL_2 & 0x10UL)))
136 {
137 tmp_swtrig |= DAC_SWTRIGR_SWTRIG2;
138 }
139 /* Enable the selected DAC software conversion*/
140 SET_BIT(hdac->Instance->SWTRIGR, tmp_swtrig);
141
142 /* Change DAC state */
143 hdac->State = HAL_DAC_STATE_READY;
144
145 /* Process unlocked */
146 __HAL_UNLOCK(hdac);
147
148 /* Return function status */
149 return HAL_OK;
150}

189HAL_StatusTypeDef HAL_DACEx_DualStart_DMA(DAC_HandleTypeDef *hdac, uint32_t Channel, uint32_t *pData, uint32_t Length,
190 uint32_t Alignment)
191{
192 HAL_StatusTypeDef status;
193 uint32_t tmpreg = 0UL;
194
195 /* Check the parameters */
196 assert_param(IS_DAC_CHANNEL(Channel));
197 assert_param(IS_DAC_ALIGN(Alignment));
198
199 /* Process locked */
200 __HAL_LOCK(hdac);
201
202 /* Change DAC state */
203 hdac->State = HAL_DAC_STATE_BUSY;
204
205 if (Channel == DAC_CHANNEL_1)
206 {
207 /* Set the DMA transfer complete callback for channel1 */
208 hdac->DMA_Handle1->XferCpltCallback = DAC_DMAConvCpltCh1;
209
210 /* Set the DMA half transfer complete callback for channel1 */
211 hdac->DMA_Handle1->XferHalfCpltCallback = DAC_DMAHalfConvCpltCh1;
212
213 /* Set the DMA error callback for channel1 */
214 hdac->DMA_Handle1->XferErrorCallback = DAC_DMAErrorCh1;
215
216 /* Enable the selected DAC channel1 DMA request */
217 SET_BIT(hdac->Instance->CR, DAC_CR_DMAEN1);
218 }
219 else
220 {
221 /* Set the DMA transfer complete callback for channel2 */
222 hdac->DMA_Handle2->XferCpltCallback = DAC_DMAConvCpltCh2;
223
224 /* Set the DMA half transfer complete callback for channel2 */
225 hdac->DMA_Handle2->XferHalfCpltCallback = DAC_DMAHalfConvCpltCh2;
226
227 /* Set the DMA error callback for channel2 */
228 hdac->DMA_Handle2->XferErrorCallback = DAC_DMAErrorCh2;
229
230 /* Enable the selected DAC channel2 DMA request */
231 SET_BIT(hdac->Instance->CR, DAC_CR_DMAEN2);
232 }
233
234 switch (Alignment)
235 {
236 case DAC_ALIGN_12B_R:
237 /* Get DHR12R1 address */
238 tmpreg = (uint32_t)&hdac->Instance->DHR12RD;
239 break;
240 case DAC_ALIGN_12B_L:
241 /* Get DHR12L1 address */
242 tmpreg = (uint32_t)&hdac->Instance->DHR12LD;
243 break;
244 case DAC_ALIGN_8B_R:
245 /* Get DHR8R1 address */

247 break;

249 break;
250 }

268 }
269
270 /* Process Unlocked */
271 __HAL_UNLOCK(hdac);
272
273 if (status == HAL_OK)
274 {
275 /* Enable the Peripheral */
276 __HAL_DAC_ENABLE(hdac, DAC_CHANNEL_1);
277 __HAL_DAC_ENABLE(hdac, DAC_CHANNEL_2);
278 }
279 else
280 {
281 hdac->ErrorCode |= HAL_DAC_ERROR_DMA;
282 }
283
284 /* Return function status */

568HAL_StatusTypeDef HAL_DACEx_SelfCalibrate(DAC_HandleTypeDef *hdac, DAC_ChannelConfTypeDef *sConfig, uint32_t Channel)
569{
570 HAL_StatusTypeDef status = HAL_OK;
571
572 __IO uint32_t tmp;
573 uint32_t trimmingvalue;
574 uint32_t delta;
575
576 /* store/restore channel configuration structure purpose */
577 uint32_t oldmodeconfiguration;
578
579 /* Check the parameters */
580 assert_param(IS_DAC_CHANNEL(Channel));
581
582 /* Check the DAC handle allocation */
583 /* Check if DAC running */
584 if (hdac == NULL)
585 {
586 status = HAL_ERROR;
587 }
588 else if (hdac->State == HAL_DAC_STATE_BUSY)
589 {
590 status = HAL_ERROR;
591 }
592 else
593 {
594 /* Process locked */
595 __HAL_LOCK(hdac);
596
597 /* Store configuration */
598 oldmodeconfiguration = (hdac->Instance->MCR & (DAC_MCR_MODE1 << (Channel & 0x10UL)));
599
600 /* Disable the selected DAC channel */
601 CLEAR_BIT((hdac->Instance->CR), (DAC_CR_EN1 << (Channel & 0x10UL)));
602
603 /* Set mode in MCR for calibration */
604 MODIFY_REG(hdac->Instance->MCR, (DAC_MCR_MODE1 << (Channel & 0x10UL)), 0U);
605
606 /* Set DAC Channel1 DHR register to the middle value */
607 tmp = (uint32_t)hdac->Instance;
608
609 if (Channel == DAC_CHANNEL_1)
610 {
611 tmp += DAC_DHR12R1_ALIGNMENT(DAC_ALIGN_12B_R);
612 }
613 else
614 {
615 tmp += DAC_DHR12R2_ALIGNMENT(DAC_ALIGN_12B_R);
616 }
617
618 *(__IO uint32_t *) tmp = 0x0800UL;
619
620 /* Enable the selected DAC channel calibration */
621 /* i.e. set DAC_CR_CENx bit */
622 SET_BIT((hdac->Instance->CR), (DAC_CR_CEN1 << (Channel & 0x10UL)));
623
624 /* Init trimming counter */
625 /* Medium value */
626 trimmingvalue = 16UL;
627 delta = 8UL;
628 while (delta != 0UL)
629 {
630 /* Set candidate trimming */
631 MODIFY_REG(hdac->Instance->CCR, (DAC_CCR_OTRIM1 << (Channel & 0x10UL)), (trimmingvalue << (Channel & 0x10UL)));
632
633 /* tOFFTRIMmax delay x ms as per datasheet (electrical characteristics */
634 /* i.e. minimum time needed between two calibration steps */
635 HAL_Delay(1);
636
637 if ((hdac->Instance->SR & (DAC_SR_CAL_FLAG1 << (Channel & 0x10UL))) == (DAC_SR_CAL_FLAG1 << (Channel & 0x10UL)))
638 {
639 /* DAC_SR_CAL_FLAGx is HIGH try higher trimming */
640 trimmingvalue -= delta;
641 }
642 else
643 {
644 /* DAC_SR_CAL_FLAGx is LOW try lower trimming */
645 trimmingvalue += delta;
646 }
647 delta >>= 1UL;
648 }
649
650 /* Still need to check if right calibration is current value or one step below */
651 /* Indeed the first value that causes the DAC_SR_CAL_FLAGx bit to change from 0 to 1 */
652 /* Set candidate trimming */
653 MODIFY_REG(hdac->Instance->CCR, (DAC_CCR_OTRIM1 << (Channel & 0x10UL)), (trimmingvalue << (Channel & 0x10UL)));
654
655 /* tOFFTRIMmax delay x ms as per datasheet (electrical characteristics */
656 /* i.e. minimum time needed between two calibration steps */
657 HAL_Delay(1U);
658
659 if ((hdac->Instance->SR & (DAC_SR_CAL_FLAG1 << (Channel & 0x10UL))) == 0UL)
660 {
661 /* Trimming is actually one value more */
662 trimmingvalue++;
663 /* Set right trimming */
664 MODIFY_REG(hdac->Instance->CCR, (DAC_CCR_OTRIM1 << (Channel & 0x10UL)), (trimmingvalue << (Channel & 0x10UL)));
665 }
666
667 /* Disable the selected DAC channel calibration */
668 /* i.e. clear DAC_CR_CENx bit */
669 CLEAR_BIT((hdac->Instance->CR), (DAC_CR_CEN1 << (Channel & 0x10UL)));
670
671 sConfig->DAC_TrimmingValue = trimmingvalue;
672 sConfig->DAC_UserTrimming = DAC_TRIMMING_USER;
673
674 /* Restore configuration */
675 MODIFY_REG(hdac->Instance->MCR, (DAC_MCR_MODE1 << (Channel & 0x10UL)), oldmodeconfiguration);
676
677 /* Process unlocked */
678 __HAL_UNLOCK(hdac);
679 }
680
681 return status;
682}
