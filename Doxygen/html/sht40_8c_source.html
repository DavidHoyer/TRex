59static result_t GetSensorData(SensorData_t *sensorData) {
60 if (sensorData == NULL)
61 return RESULT_INVALID_ARGUMENTS;
62
63 R_TRY(I2C_Write(DEVICE_ADDRESS, Precision_High));
64 CORE_Delay(10);
65
66 struct {
67 uint8_t th;
68 uint8_t tl;
69 uint8_t tcrc8;
70 uint8_t rhh;
71 uint8_t rhl;
72 uint8_t rhcrc8;
73 } raw_data = { 0 };
74
75 R_TRY(I2C_ReadBuffer(DEVICE_ADDRESS, (uint8_t*)&raw_data, sizeof(raw_data)));
76
77 sensorData->humidity = -6 + 125 * (float32_t)((((uint16_t)raw_data.rhh) << 8) | raw_data.rhl) / 0xFFFF;
78 sensorData->temperature = -45 + 175 * (float32_t)((((uint16_t)raw_data.th) << 8) | raw_data.tl) / 0xFFFF;
79
80 return RESULT_SUCCESS;
81}
