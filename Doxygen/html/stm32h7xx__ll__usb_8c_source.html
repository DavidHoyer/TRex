82HAL_StatusTypeDef USB_CoreInit(USB_OTG_GlobalTypeDef *USBx, USB_OTG_CfgTypeDef cfg)
83{
84 HAL_StatusTypeDef ret;
85
86 if (cfg.phy_itface == USB_OTG_ULPI_PHY)
87 {

89
90 /* Init The ULPI Interface */

92
93 /* Select vbus source */
94 USBx->GUSBCFG &= ~(USB_OTG_GUSBCFG_ULPIEVBUSD | USB_OTG_GUSBCFG_ULPIEVBUSI);

96 {
97 USBx->GUSBCFG |= USB_OTG_GUSBCFG_ULPIEVBUSD;
98 }
99
100 /* Reset after a PHY select */
101 ret = USB_CoreReset(USBx);
102 }
103 else /* FS interface (embedded Phy) */
104 {
105 /* Select FS Embedded PHY */
106 USBx->GUSBCFG |= USB_OTG_GUSBCFG_PHYSEL;
107
108 /* Reset after a PHY select */
109 ret = USB_CoreReset(USBx);
110
111 if (cfg.battery_charging_enable == 0U)
112 {
113 /* Activate the USB Transceiver */
114 USBx->GCCFG |= USB_OTG_GCCFG_PWRDWN;
115 }
116 else
117 {
118 /* Deactivate the USB Transceiver */
119 USBx->GCCFG &= ~(USB_OTG_GCCFG_PWRDWN);
120 }
121 }
122
123 if (cfg.dma_enable == 1U)
124 {
125 /* make sure to reserve 18 fifo Locations for DMA buffers */
126 USBx->GDFIFOCFG &= ~(0xFFFFU << 16);
127 USBx->GDFIFOCFG |= 0x3EEU << 16;
128
129 USBx->GAHBCFG |= USB_OTG_GAHBCFG_HBSTLEN_2;
130 USBx->GAHBCFG |= USB_OTG_GAHBCFG_DMAEN;
131 }
132
133 return ret;
134}

143HAL_StatusTypeDef USB_SetTurnaroundTime(USB_OTG_GlobalTypeDef *USBx,
144 uint32_t hclk, uint8_t speed)
145{
146 uint32_t UsbTrd;
147
148 /* The USBTRD is configured according to the tables below, depending on AHB frequency
149 used by application. In the low AHB frequency range it is used to stretch enough the USB response
150 time to IN tokens, the USB turnaround time, so to compensate for the longer AHB read access
151 latency to the Data FIFO */
152 if (speed == USBD_FS_SPEED)
153 {
154 if ((hclk >= 14200000U) && (hclk < 15000000U))
155 {
156 /* hclk Clock Range between 14.2-15 MHz */
157 UsbTrd = 0xFU;
158 }
159 else if ((hclk >= 15000000U) && (hclk < 16000000U))
160 {
161 /* hclk Clock Range between 15-16 MHz */
162 UsbTrd = 0xEU;
163 }
164 else if ((hclk >= 16000000U) && (hclk < 17200000U))
165 {
166 /* hclk Clock Range between 16-17.2 MHz */
167 UsbTrd = 0xDU;
168 }
169 else if ((hclk >= 17200000U) && (hclk < 18500000U))
170 {
171 /* hclk Clock Range between 17.2-18.5 MHz */
172 UsbTrd = 0xCU;
173 }
174 else if ((hclk >= 18500000U) && (hclk < 20000000U))
175 {
176 /* hclk Clock Range between 18.5-20 MHz */
177 UsbTrd = 0xBU;
178 }
179 else if ((hclk >= 20000000U) && (hclk < 21800000U))
180 {
181 /* hclk Clock Range between 20-21.8 MHz */
182 UsbTrd = 0xAU;
183 }
184 else if ((hclk >= 21800000U) && (hclk < 24000000U))
185 {
186 /* hclk Clock Range between 21.8-24 MHz */
187 UsbTrd = 0x9U;
188 }
189 else if ((hclk >= 24000000U) && (hclk < 27700000U))
190 {
191 /* hclk Clock Range between 24-27.7 MHz */
192 UsbTrd = 0x8U;
193 }
194 else if ((hclk >= 27700000U) && (hclk < 32000000U))
195 {
196 /* hclk Clock Range between 27.7-32 MHz */
197 UsbTrd = 0x7U;
198 }
199 else /* if(hclk >= 32000000) */
200 {
201 /* hclk Clock Range between 32-200 MHz */
202 UsbTrd = 0x6U;
203 }
204 }
205 else if (speed == USBD_HS_SPEED)
206 {
207 UsbTrd = USBD_HS_TRDT_VALUE;
208 }
209 else
210 {
211 UsbTrd = USBD_DEFAULT_TRDT_VALUE;
212 }
213
214 USBx->GUSBCFG &= ~USB_OTG_GUSBCFG_TRDT;
215 USBx->GUSBCFG |= (uint32_t)((UsbTrd << 10) & USB_OTG_GUSBCFG_TRDT);
216
217 return HAL_OK;
218}

282HAL_StatusTypeDef USB_DevInit(USB_OTG_GlobalTypeDef *USBx, USB_OTG_CfgTypeDef cfg)
283{
284 HAL_StatusTypeDef ret = HAL_OK;
285 uint32_t USBx_BASE = (uint32_t)USBx;
286 uint32_t i;
287
288 for (i = 0U; i < 15U; i++)
289 {
290 USBx->DIEPTXF[i] = 0U;
291 }
292
293 /* VBUS Sensing setup */
294 if (cfg.vbus_sensing_enable == 0U)
295 {
296 USBx_DEVICE->DCTL |= USB_OTG_DCTL_SDIS;
297
298 /* Deactivate VBUS Sensing B */
299 USBx->GCCFG &= ~USB_OTG_GCCFG_VBDEN;
300
301 /* B-peripheral session valid override enable */
302 USBx->GOTGCTL |= USB_OTG_GOTGCTL_BVALOEN;
303 USBx->GOTGCTL |= USB_OTG_GOTGCTL_BVALOVAL;
304 }
305 else
306 {
307 /* Enable HW VBUS sensing */
308 USBx->GCCFG |= USB_OTG_GCCFG_VBDEN;
309 }
310
311 /* Restart the Phy Clock */
312 USBx_PCGCCTL = 0U;
313
314 /* Device mode configuration */
315 USBx_DEVICE->DCFG |= DCFG_FRAME_INTERVAL_80;
316
317 if (cfg.phy_itface == USB_OTG_ULPI_PHY)
318 {
319 if (cfg.speed == USBD_HS_SPEED)
320 {
321 /* Set Core speed to High speed mode */
322 (void)USB_SetDevSpeed(USBx, USB_OTG_SPEED_HIGH);
323 }
324 else
325 {
326 /* Set Core speed to Full speed mode */
327 (void)USB_SetDevSpeed(USBx, USB_OTG_SPEED_HIGH_IN_FULL);
328 }
329 }
330 else
331 {
332 /* Set Core speed to Full speed mode */
333 (void)USB_SetDevSpeed(USBx, USB_OTG_SPEED_FULL);
334 }
335
336 /* Flush the FIFOs */
337 if (USB_FlushTxFifo(USBx, 0x10U) != HAL_OK) /* all Tx FIFOs */
338 {
339 ret = HAL_ERROR;
340 }
341
342 if (USB_FlushRxFifo(USBx) != HAL_OK)
343 {
344 ret = HAL_ERROR;
345 }
346
347 /* Clear all pending Device Interrupts */
348 USBx_DEVICE->DIEPMSK = 0U;
349 USBx_DEVICE->DOEPMSK = 0U;
350 USBx_DEVICE->DAINTMSK = 0U;
351
352 for (i = 0U; i < cfg.dev_endpoints; i++)
353 {
354 if ((USBx_INEP(i)->DIEPCTL & USB_OTG_DIEPCTL_EPENA) == USB_OTG_DIEPCTL_EPENA)
355 {
356 if (i == 0U)
357 {
358 USBx_INEP(i)->DIEPCTL = USB_OTG_DIEPCTL_SNAK;
359 }
360 else
361 {
362 USBx_INEP(i)->DIEPCTL = USB_OTG_DIEPCTL_EPDIS | USB_OTG_DIEPCTL_SNAK;
363 }
364 }
365 else
366 {
367 USBx_INEP(i)->DIEPCTL = 0U;
368 }
369
370 USBx_INEP(i)->DIEPTSIZ = 0U;
371 USBx_INEP(i)->DIEPINT = 0xFB7FU;
372 }
373
374 for (i = 0U; i < cfg.dev_endpoints; i++)
375 {
376 if ((USBx_OUTEP(i)->DOEPCTL & USB_OTG_DOEPCTL_EPENA) == USB_OTG_DOEPCTL_EPENA)
377 {
378 if (i == 0U)
379 {
380 USBx_OUTEP(i)->DOEPCTL = USB_OTG_DOEPCTL_SNAK;
381 }
382 else
383 {
384 USBx_OUTEP(i)->DOEPCTL = USB_OTG_DOEPCTL_EPDIS | USB_OTG_DOEPCTL_SNAK;
385 }
386 }
387 else
388 {
389 USBx_OUTEP(i)->DOEPCTL = 0U;
390 }
391
392 USBx_OUTEP(i)->DOEPTSIZ = 0U;
393 USBx_OUTEP(i)->DOEPINT = 0xFB7FU;
394 }
395
396 USBx_DEVICE->DIEPMSK &= ~(USB_OTG_DIEPMSK_TXFURM);
397
398 /* Disable all interrupts. */
399 USBx->GINTMSK = 0U;
400
401 /* Clear any pending interrupts */
402 USBx->GINTSTS = 0xBFFFFFFFU;
403
404 /* Enable the common interrupts */
405 if (cfg.dma_enable == 0U)
406 {
407 USBx->GINTMSK |= USB_OTG_GINTMSK_RXFLVLM;
408 }
409
410 /* Enable interrupts matching to the Device mode ONLY */
411 USBx->GINTMSK |= USB_OTG_GINTMSK_USBSUSPM | USB_OTG_GINTMSK_USBRST |
412 USB_OTG_GINTMSK_ENUMDNEM | USB_OTG_GINTMSK_IEPINT |
413 USB_OTG_GINTMSK_OEPINT | USB_OTG_GINTMSK_IISOIXFRM |
414 USB_OTG_GINTMSK_PXFRM_IISOOXFRM | USB_OTG_GINTMSK_WUIM;
415
416 if (cfg.Sof_enable != 0U)
417 {
418 USBx->GINTMSK |= USB_OTG_GINTMSK_SOFM;
419 }
420
421 if (cfg.vbus_sensing_enable == 1U)
422 {
423 USBx->GINTMSK |= (USB_OTG_GINTMSK_SRQIM | USB_OTG_GINTMSK_OTGINT);
424 }
425
426 return ret;
427}

698HAL_StatusTypeDef USB_EPStartXfer(USB_OTG_GlobalTypeDef *USBx, USB_OTG_EPTypeDef *ep, uint8_t dma)
699{
700 uint32_t USBx_BASE = (uint32_t)USBx;
701 uint32_t epnum = (uint32_t)ep->num;
702 uint16_t pktcnt;
703
704 /* IN endpoint */
705 if (ep->is_in == 1U)
706 {
707 /* Zero Length Packet? */
708 if (ep->xfer_len == 0U)
709 {
710 USBx_INEP(epnum)->DIEPTSIZ &= ~(USB_OTG_DIEPTSIZ_PKTCNT);
711 USBx_INEP(epnum)->DIEPTSIZ |= (USB_OTG_DIEPTSIZ_PKTCNT & (1U << 19));
712 USBx_INEP(epnum)->DIEPTSIZ &= ~(USB_OTG_DIEPTSIZ_XFRSIZ);
713 }
714 else
715 {
716 /* Program the transfer size and packet count
717 * as follows: xfersize = N * maxpacket +
718 * short_packet pktcnt = N + (short_packet
719 * exist ? 1 : 0)
720 */
721 USBx_INEP(epnum)->DIEPTSIZ &= ~(USB_OTG_DIEPTSIZ_XFRSIZ);
722 USBx_INEP(epnum)->DIEPTSIZ &= ~(USB_OTG_DIEPTSIZ_PKTCNT);
723 USBx_INEP(epnum)->DIEPTSIZ |= (USB_OTG_DIEPTSIZ_PKTCNT & (((ep->xfer_len + ep->maxpacket - 1U) / ep->maxpacket) << 19));
724 USBx_INEP(epnum)->DIEPTSIZ |= (USB_OTG_DIEPTSIZ_XFRSIZ & ep->xfer_len);
725
726 if (ep->type == EP_TYPE_ISOC)
727 {
728 USBx_INEP(epnum)->DIEPTSIZ &= ~(USB_OTG_DIEPTSIZ_MULCNT);
729 USBx_INEP(epnum)->DIEPTSIZ |= (USB_OTG_DIEPTSIZ_MULCNT & (1U << 29));
730 }
731 }
732
733 if (dma == 1U)
734 {
735 if ((uint32_t)ep->dma_addr != 0U)
736 {
737 USBx_INEP(epnum)->DIEPDMA = (uint32_t)(ep->dma_addr);
738 }
739
740 if (ep->type == EP_TYPE_ISOC)
741 {
742 if ((USBx_DEVICE->DSTS & (1U << 8)) == 0U)
743 {
744 USBx_INEP(epnum)->DIEPCTL |= USB_OTG_DIEPCTL_SODDFRM;
745 }
746 else
747 {
748 USBx_INEP(epnum)->DIEPCTL |= USB_OTG_DIEPCTL_SD0PID_SEVNFRM;
749 }
750 }
751
752 /* EP enable, IN data in FIFO */
753 USBx_INEP(epnum)->DIEPCTL |= (USB_OTG_DIEPCTL_CNAK | USB_OTG_DIEPCTL_EPENA);
754 }
755 else
756 {
757 /* EP enable, IN data in FIFO */
758 USBx_INEP(epnum)->DIEPCTL |= (USB_OTG_DIEPCTL_CNAK | USB_OTG_DIEPCTL_EPENA);
759
760 if (ep->type != EP_TYPE_ISOC)
761 {
762 /* Enable the Tx FIFO Empty Interrupt for this EP */
763 if (ep->xfer_len > 0U)
764 {
765 USBx_DEVICE->DIEPEMPMSK |= 1UL << (ep->num & EP_ADDR_MSK);
766 }
767 }
768 else
769 {
770 if ((USBx_DEVICE->DSTS & (1U << 8)) == 0U)
771 {
772 USBx_INEP(epnum)->DIEPCTL |= USB_OTG_DIEPCTL_SODDFRM;
773 }
774 else
775 {
776 USBx_INEP(epnum)->DIEPCTL |= USB_OTG_DIEPCTL_SD0PID_SEVNFRM;
777 }
778
779 (void)USB_WritePacket(USBx, ep->xfer_buff, ep->num, (uint16_t)ep->xfer_len, dma);
780 }
781 }
782 }
783 else /* OUT endpoint */
784 {
785 /* Program the transfer size and packet count as follows:
786 * pktcnt = N
787 * xfersize = N * maxpacket
788 */
789 USBx_OUTEP(epnum)->DOEPTSIZ &= ~(USB_OTG_DOEPTSIZ_XFRSIZ);
790 USBx_OUTEP(epnum)->DOEPTSIZ &= ~(USB_OTG_DOEPTSIZ_PKTCNT);
791

793 {
794 USBx_OUTEP(epnum)->DOEPTSIZ |= (USB_OTG_DOEPTSIZ_XFRSIZ & ep->maxpacket);
795 USBx_OUTEP(epnum)->DOEPTSIZ |= (USB_OTG_DOEPTSIZ_PKTCNT & (1U << 19));
796 }
797 else
798 {
799 pktcnt = (uint16_t)((ep->xfer_len + ep->maxpacket - 1U) / ep->maxpacket);
800 USBx_OUTEP(epnum)->DOEPTSIZ |= USB_OTG_DOEPTSIZ_PKTCNT & ((uint32_t)pktcnt << 19);
801 USBx_OUTEP(epnum)->DOEPTSIZ |= USB_OTG_DOEPTSIZ_XFRSIZ & (ep->maxpacket * pktcnt);
802 }
803
804 if (dma == 1U)
805 {
806 if ((uint32_t)ep->xfer_buff != 0U)
807 {
808 USBx_OUTEP(epnum)->DOEPDMA = (uint32_t)(ep->xfer_buff);
809 }
810 }
811
812 if (ep->type == EP_TYPE_ISOC)
813 {
814 if ((USBx_DEVICE->DSTS & (1U << 8)) == 0U)
815 {
816 USBx_OUTEP(epnum)->DOEPCTL |= USB_OTG_DOEPCTL_SODDFRM;
817 }
818 else
819 {
820 USBx_OUTEP(epnum)->DOEPCTL |= USB_OTG_DOEPCTL_SD0PID_SEVNFRM;
821 }
822 }
823 /* EP enable */
824 USBx_OUTEP(epnum)->DOEPCTL |= (USB_OTG_DOEPCTL_CNAK | USB_OTG_DOEPCTL_EPENA);
825 }
826
827 return HAL_OK;
828}

1711HAL_StatusTypeDef USB_HC_StartXfer(USB_OTG_GlobalTypeDef *USBx, USB_OTG_HCTypeDef *hc, uint8_t dma)
1712{

1714 uint32_t ch_num = (uint32_t)hc->ch_num;
1715 __IO uint32_t tmpreg;
1716 uint8_t is_oddframe;
1717 uint16_t len_words;
1718 uint16_t num_packets;
1719 uint16_t max_hc_pkt_count = 256U;
1720
1721 if (((USBx->CID & (0x1U << 8)) != 0U) && (hc->speed == USBH_HS_SPEED))
1722 {
1723 /* in DMA mode host Core automatically issues ping in case of NYET/NAK */
1724 if ((dma == 1U) && ((hc->ep_type == EP_TYPE_CTRL) || (hc->ep_type == EP_TYPE_BULK)))
1725 {
1726 USBx_HC((uint32_t)ch_num)->HCINTMSK &= ~(USB_OTG_HCINTMSK_NYET |
1727 USB_OTG_HCINTMSK_ACKM |

1732 {

1735 }
1736
1737 }
1738

1740 if (hc->xfer_len > 0U)
1741 {
1742 num_packets = (uint16_t)((hc->xfer_len + hc->max_packet - 1U) / hc->max_packet);
1743

1745 {
1746 num_packets = max_hc_pkt_count;

1756 * For IN channel HCTSIZ.XferSize is expected to be an integer multiple of

1758 */

1761 hc->XferSize = (uint32_t)num_packets * hc->max_packet;
1762 }
1763 else
1764 {

1766 }

1770 (((uint32_t)num_packets << 19) & USB_OTG_HCTSIZ_PKTCNT) |
1771 (((uint32_t)hc->data_pid << 29) & USB_OTG_HCTSIZ_DPID);
1772
1773 if (dma != 0U)
1774 {
1775 /* xfer_buff MUST be 32-bits aligned */
1776 USBx_HC(ch_num)->HCDMA = (uint32_t)hc->xfer_buff;
1777 }

1780 USBx_HC(ch_num)->HCCHAR &= ~USB_OTG_HCCHAR_ODDFRM;

1784 tmpreg = USBx_HC(ch_num)->HCCHAR;
1785 tmpreg &= ~USB_OTG_HCCHAR_CHDIS;
1786
1787 /* make sure to set the correct ep direction */
1788 if (hc->ep_is_in != 0U)
1789 {
1790 tmpreg |= USB_OTG_HCCHAR_EPDIR;
1791 }
1792 else
1793 {
1794 tmpreg &= ~USB_OTG_HCCHAR_EPDIR;
1795 }
1796 tmpreg |= USB_OTG_HCCHAR_CHENA;
1797 USBx_HC(ch_num)->HCCHAR = tmpreg;
1798

1800 {
1801 return HAL_OK;
1802 }
1803
1804 if ((hc->ep_is_in == 0U) && (hc->xfer_len > 0U))
1805 {
1806 switch (hc->ep_type)
1807 {
1808 /* Non periodic transfer */
1809 case EP_TYPE_CTRL:
1810 case EP_TYPE_BULK:
1811
1812 len_words = (uint16_t)((hc->xfer_len + 3U) / 4U);
1813
1814 /* check if there is enough space in FIFO space */
1815 if (len_words > (USBx->HNPTXSTS & 0xFFFFU))
1816 {
1817 /* need to process data in nptxfempty interrupt */
1818 USBx->GINTMSK |= USB_OTG_GINTMSK_NPTXFEM;
1819 }
1820 break;
1821
1822 /* Periodic transfer */
1823 case EP_TYPE_INTR:
1824 case EP_TYPE_ISOC:
1825 len_words = (uint16_t)((hc->xfer_len + 3U) / 4U);
1826 /* check if there is enough space in FIFO space */
1827 if (len_words > (USBx_HOST->HPTXSTS & 0xFFFFU)) /* split the transfer */
1828 {
1829 /* need to process data in ptxfempty interrupt */
1830 USBx->GINTMSK |= USB_OTG_GINTMSK_PTXFEM;
1831 }
1832 break;
1833
1834 default:
1835 break;
1836 }
1837
1838 /* Write packet into the Tx FIFO. */
1839 (void)USB_WritePacket(USBx, hc->xfer_buff, hc->ch_num, (uint16_t)hc->xfer_len, 0);
1840 }
1841
1842 return HAL_OK;
1843}
