324{
325 uint16_t _SwapVal, _Byte1, _Byte2;
326 uint8_t *_pbuff = addr;
327
328 _Byte1 = *(uint8_t *)_pbuff;
329 _pbuff++;
330 _Byte2 = *(uint8_t *)_pbuff;

332 _SwapVal = (_Byte2 << 8) | _Byte1;
