74result_t LCD_SetDrawArea(uint16_t x1, uint16_t y1, uint16_t x2, uint16_t y2) {
75 if (x1 > LCD_WIDTH || y1 > LCD_HEIGHT)
76 return RESULT_INVALID_ARGUMENTS;
77 if (x2 > LCD_WIDTH || y2 > LCD_HEIGHT)
78 return RESULT_INVALID_ARGUMENTS;
79
80 /* Set horizontal region */
81 LCD_Command(LCD_CMD_SET_COLUMN_ADDRESS);
82 LCD_Data8(x1 >> 8);
83 LCD_Data8(x1 & 0xFF);
84 LCD_Data8(x2 >> 8);

87 /* Set vertical region */
88 LCD_Command(LCD_CMD_SET_PAGE_ADDRESS);
89 LCD_Data8(y1 >> 8);
90 LCD_Data8(y1 & 0xFF);
91 LCD_Data8(y2 >> 8);
92 LCD_Data8(y2 & 0xFF);
93
94 return RESULT_SUCCESS;
95}

101result_t LCD_Init(void) {
102
103 /* Turn display on */
104 LCD_Command(LCD_CMD_SET_DISPLAY_ON);
105
106 /* Flip LCD horizontally */
107 LCD_CommandData8(LCD_CMD_SET_ADDRESS_MODE, 0b00000010);
108
109 CORE_Delay(10);
110
111 /* Set LCD color format */
112 LCD->cmd = LCD_CMD_SET_PIXEL_DATA_INTERFACE;
113 LCD_CommandData8(LCD_CMD_SET_PIXEL_DATA_INTERFACE, 0b011); // RGB565
114
115 LCD_SetDrawArea(0, 0, LCD_WIDTH - 1, LCD_HEIGHT - 1);
116
117 CORE_Delay(100);
118
119 /* Start pixel access mode */
120 LCD_EnableDrawMode();
121
122 LCD_SetForegroundColor(ColorWhite);
123 LCD_SetBackgroundColor(ColorBlack);
124 LCD_Clear();
125
126 return RESULT_SUCCESS;
127}

186result_t LCD_Line(uint16_t x1, uint16_t y1, uint16_t x2, uint16_t y2) {
187 if (x1 > LCD_WIDTH || y1 > LCD_HEIGHT)
188 return RESULT_INVALID_ARGUMENTS;
189 if (x2 > LCD_WIDTH || y2 > LCD_HEIGHT)
190 return RESULT_INVALID_ARGUMENTS;
191
192 int16_t delta_x = +abs((int32_t)x2 - (int32_t)x1);
193 int16_t delta_y = -abs((int32_t)y2 - (int32_t)y1);
194
195 int16_t speed_x = x1 < x2 ? 1 : -1;
196 int16_t speed_y = y1 < y2 ? 1 : -1;
197
198 int16_t error = delta_x + delta_y;
199
200 /* Bresenham's line drawing algorithm */
201 while (x1 != x2 && y1 != y2) {
202 LCD_Pixel(x1, y1);
203
204 if ((error * 2) > delta_y) {
205 error += delta_y;
206 x1 += speed_x;
207 }
208
209 if ((error * 2) < delta_x) {
210 error += delta_x;
211 y1 += speed_y;
212 }
213 }
214
215 return RESULT_SUCCESS;
216}

230result_t LCD_Circle(uint16_t center_x, uint16_t center_y, uint16_t radius) {
231 /* Explicitly not check for invalid positions here because
232 * placing a circle outside of the viewing window might
233 * be useful to draw e.g semi-circles
234 */
235
236 int16_t x = 0;
237 int16_t y = radius;
238 int16_t d = 3 - 2 * radius;
239
240 LCD_CircleImpl(center_x, center_y, x, y);
241
242 /* Bresenham's circle drawing algorithm */
243 while (y >= x) {
244 x += 1;
245
246 if (d > 0) {
247 y -= 1;
248 d += 4 * (x - y) + 10;
249 } else {
250 d += 4 * x + 6;
251 }
252
253 LCD_CircleImpl(center_x, center_y, x, y);
254 }
255
256 return RESULT_SUCCESS;
257}

259result_t LCD_Character(uint16_t x, uint16_t y, char c) {
260 /* Set working region same size as character bitmap */
261 R_TRY(LCD_SetDrawArea(x, y, x + CONSOLE_FONT_CHAR_WIDTH - 1, y + CONSOLE_FONT_CHAR_HEIGHT - 1));
262 LCD_EnableDrawMode();
263
264 /* Draw all 1s in bitmap with foreground color, all 0s with background color */
265 for (uint32_t i = 0; i < CONSOLE_FONT_CHAR_WIDTH * CONSOLE_FONT_CHAR_HEIGHT; i++) {
266 if (ConsoleFont[(uint8_t)c][i / 8] & (1 << (i % 8)))
267 LCD_Set(&m_foreground_color);
268 else {
269 LCD_Set(&m_background_color);
270 }
271 }
272
273 return RESULT_SUCCESS;
274}

276result_t LCD_String(uint16_t start_x, uint16_t start_y, const char *string) {
277 uint16_t x = start_x, y = start_y;
278 while (true) {
279 char c = *string++;
280
281 if (c == '\n') {
282 x = 0;
283 y += CONSOLE_FONT_CHAR_HEIGHT;
284 continue;
285 } else if (c == '\x00') {
286 break;
287 }
288
289 if (LCD_Character(x, y, c) != RESULT_SUCCESS)
290 return RESULT_INVALID_ARGUMENTS;
291
292 /* Advance cursor horizontally */
293 x += CONSOLE_FONT_CHAR_WIDTH;
294 }
295
296 m_console_cursor_x = x;
297 m_console_cursor_y = y;
298
299 return RESULT_SUCCESS;
300}

426result_t LCD_TouchGetPosition(LCD_TouchPosition_t *position) {
427 if (position == NULL)
428 return RESULT_INVALID_ARGUMENTS;
429
430 uint8_t send_buffer[5] = { (LCD_TOUCH_CMD_MEASURE_X << 1) | 1, 0, 0, 0, 0 };
431 uint8_t recv_buffer[5] = { 0 };
432
433 do {
434 LCD_TouchStartMeasurement();
435 LCD_TouchEnable();
436 CORE_Delay(3);
437
438 memset(recv_buffer, 0x00, 5);
439
440 R_TRY(SPI_WriteRead(SPI_Touch, recv_buffer, send_buffer, 5));
441
442 LCD_TouchDisable();
443 } while ((recv_buffer[2] & 0b11) == 0b11 || (recv_buffer[4] & 0b11) == 0b11);
444
445 LCD_TouchStartMeasurement();
446
447 if ((recv_buffer[2] & 0b11) != 0b00 || (recv_buffer[4] & 0b11) != 0b00) {
448 position->x = 0;
449 position->y = 0;
450 } else {
451 position->x = recv_buffer[1] << 4 | recv_buffer[2] >> 4;
452 position->y = recv_buffer[3] << 4 | recv_buffer[4] >> 4;
453 }
454
455 position->x = ((float32_t)position->x / 0x1000) * LCD_WIDTH;
456 position->y = ((float32_t)(0x1000 - position->y) / 0x1000) * LCD_HEIGHT;
457
458 return RESULT_SUCCESS;
459}
