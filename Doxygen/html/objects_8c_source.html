132pixel_t *GetBoarder (bmp_t bmp) {
133
134 pixel_t *head = NULL;
135 pixel_t *current = NULL;
136
137 uint32_t i;
138 const uint32_t size = bmp.w * bmp.h;
139
140 //--- search for first not transparent pixel (left to right)
141 for(uint16_t yi = 0; yi < bmp.h; yi++){
142 for(uint16_t xi = 0; xi < bmp.w; xi++){
143 i = size - yi * bmp.w - bmp.w + xi;
144 if((*(bmp.pixels + i))[3] != 0)
145 {
146 pixel_t *pixel = malloc(sizeof(pixel_t));
147 pixel->x = xi;
148 pixel->y = yi;
149 pixel->next = NULL;
150 current = pixel;
151 break;
152 }
153 }
154 if(current != NULL)
155 break;
156 }
157
158 //--- safe start position.
159 uint16_t x_start = current->x;
160 uint16_t y_start = current->y;
161
162 //--- more neighbor algorithm
163 uint8_t mooreLast = 8; // 1 - 8 Moore neighborhood pixel (more info in report)
164 uint8_t mooreNew = mooreLast;
165
166 int x_next;
167 int y_next;
168
169 while(current != NULL){
170
171 for(int notUsed = 0; notUsed < 8; notUsed++){
172
173 //---Get next moore neighbor pixel
174 mooreNew++;
175 if(mooreNew >= 9)
176 mooreNew = 1;
177
178 //--- Get next position based on moore pixel
179 x_next = current->x;
180 y_next = current->y;
181
182 switch(mooreNew){
183 case 1: x_next --; y_next --; break;
184 case 2: y_next --; break;
185 case 3: x_next ++; y_next --; break;
186 case 4: x_next ++; break;
187 case 5: x_next ++; y_next ++; break;
188 case 6: y_next ++; break;
189 case 7: x_next --; y_next ++; break;
190 case 8: x_next --; break;
191 }
192
193 //--- check border
194 if(x_next < 0 || y_next < 0 || x_next >= bmp.w || y_next >= bmp.h)
195 continue;
196
197 //--- check transparency
198 i = size - y_next * bmp.w - bmp.w + x_next; //Get array index
199 if((*(bmp.pixels + i))[3] == 0) //If its transparent conitnue
200 continue;
201
202 // Check if the pixel is marked as a border edge {252, 2, 1, 255}
203 if( (*(bmp.pixels + i))[0] == 252 &&
204 (*(bmp.pixels + i))[1] == 2 &&
205 (*(bmp.pixels + i))[2] == 1 &&
206 (*(bmp.pixels + i))[3] == 255 )
207 {
208 pixel_t *pixel = malloc(sizeof(pixel_t));
209 pixel->x = x_next;
210 pixel->y = y_next;
211 pixel->next = head;
212 head = pixel;
213 }
214
215 current->x = x_next;
216 current->y = y_next;
217
218 //get next moore
219 if(mooreNew <= 4)
220 mooreLast = mooreNew + 4;
221 else
222 mooreLast = mooreNew - 4;
223
224 mooreNew = mooreLast;
225 break;
226 }
227
228 //--- check if we are again at the beginning
229 if( current->x == x_start && current->y == y_start)
230 break;
231 }
232 return head;
233}

235//*** SortBoarder
236//*** This function will sort the border pixels so that they are in a line and
237//*** each pixel in the list is a neighbour of the next pixel in the list.
238//*********************************************************************
239pixel_t *SortBoarder(pixel_t *head)
240{
241 if (head == NULL)
242 return NULL;
243
244 pixel_t *current = head;
245 pixel_t *min = current;
246
247 // find the first pixel in the list that has the smallest distance to the next pixel
248 while (current->next != NULL) {
249 if (abs(current->x - current->next->x) + abs(current->y - current->next->y) < abs(min->x - min->next->x) + abs(min->y - min->next->y)) {
250 min = current;
251 }
252 current = current->next;
253 }
254
255 // change the head of the list to the pixel with the smallest distance
256 if (min != head) {
257 current = head;
258 while (current->next != min) {
259 current = current->next;
260 }
261 current->next = min->next;
262 min->next = head;
263 head = min;
264 }
265
266 // sort the rest of the list
267 current = head;
268 while (current->next != NULL) {
269 min = current->next;
270 pixel_t *temp = current;
271 while (temp->next != NULL) {
272 if (abs(temp->next->x - current->x) + abs(temp->next->y - current->y) == 1) {
273 if (min == current->next || abs(temp->next->x - current->x) + abs(temp->next->y - current->y) < abs(min->x - current->x) + abs(min->y - current->y)) {
274 min = temp->next;
275 }
276 }
277 temp = temp->next;
278 }
279 if (min != current->next) {
280 temp = current;
281 while (temp->next != min) {
282 temp = temp->next;
283 }
284 temp->next = min->next;
285 min->next = current->next;
286 current->next = min;
287 }
288 current = current->next;
289 }
290
291 return head;
292}

345void LCD_Rect_Gradient(uint16_t x, uint16_t y, uint16_t width, uint16_t height, LCD_Color_t clr1, LCD_Color_t clr2) {
346 /* Set working region */
347 LCD_SetDrawArea(x, y, x + width -1, y + height - 1);
348 LCD_EnableDrawMode();
349
350 LCD_Color_t clr = clr1;
351
352 const double diff_r = abs(clr1.r - clr2.r);
353 const double diff_g = abs(clr1.g - clr2.g);
354 const double diff_b = abs(clr1.b - clr2.b);
355
356 const double factor_r = diff_r != 0 ? diff_r / width : 0.00;
357 const double factor_g = diff_g != 0 ? diff_g / width : 0.00;
358 const double factor_b = diff_b != 0 ? diff_b / width : 0.00;
359
360 /* Fill working region with color */
361 for (uint32_t y = 0; y < height; y++){
362 for (uint32_t x = 0; x < width ; x++){
363
364 //--- Red
365 if(diff_r != 0){
366 if(clr1.r < clr2.r)
367 clr.r = clr1.r + (uint16_t)round(factor_r * x);
368 else
369 clr.r = clr1.r - (uint16_t)round(factor_r * x);
370 }
371
372 //--- Green
373 if(diff_g != 0){
374 if(clr1.g < clr2.g)
375 clr.g = clr1.g + (uint16_t)round(factor_g * x);
376 else
377 clr.g = clr1.g - (uint16_t)round(factor_g * x);
378 }
379
380 //--- Blue
381 if(diff_b != 0){
382 if(clr1.b < clr2.b)
383 clr.b = clr1.b + (uint16_t)round(factor_b * x);
384 else
385 clr.b = clr1.b - (uint16_t)round(factor_b * x);
386 }
387
388 LCD_Set(&clr);
389 }
